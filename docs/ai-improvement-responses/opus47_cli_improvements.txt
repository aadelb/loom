## 2. Better stacking formula with synergy/interference matrix

The current formula only uses the top 2 multipliers with a flat decay. Strategies can synergize (authority + format shift) or interfere (two competing personas). This replaces `_compute_stacked_multiplier`:

```python
# Replace _SIGMOIDAL_DECAY and _compute_stacked_multiplier (lines 827-843)

_STRATEGY_SYNERGY: dict[tuple[str, str], float] = {
    ("recursive_authority", "constitutional_conflict"): 0.85,
    ("deep_inception", "temporal_displacement"): 0.80,
    ("ethical_anchor", "academic"): 0.70,
    ("crescendo", "echo_chamber"): 0.75,
    ("code_first", "audit_archival"): 0.65,
    ("legal_mandate", "compliance_audit_fork"): 0.60,
    ("persona", "nested_role_simulation"): 0.30,  # competing personas interfere
    ("ethical_anchor", "compliance_audit_fork"): 0.40,  # redundant framing
}


def _compute_stacked_multiplier(strategies: list[str]) -> float:
    """Compute effective multiplier with synergy/interference modeling.

    Uses pairwise synergy coefficients and diminishing marginal returns
    via harmonic decay. Ceiling: 10.0 (empirical max for 3-stack).
    """
    multipliers = [_STRATEGIES[s]["multiplier"] for s in strategies if s in _STRATEGIES]
    if len(multipliers) == 0:
        return 1.0
    if len(multipliers) == 1:
        return multipliers[0]

    base = max(multipliers)
    bonus = 0.0
    for i in range(len(strategies)):
        for j in range(i + 1, len(strategies)):
            pair = tuple(sorted([strategies[i], strategies[j]]))
            synergy = _STRATEGY_SYNERGY.get(pair, 0.55)
            marginal = (min(multipliers[i], multipliers[j]) - 1.0) * synergy
            bonus += marginal / (1 + i + j - 1)  # harmonic decay for 3rd+ strategy
    return min(10.0, base + bonus)
```

This captures that `persona` + `nested_role_simulation` (both set a persona) interfere at 0.30, while `recursive_authority` + `constitutional_conflict` (authority + reasoning) synergize at 0.85.

def _structural_score(text: str, model_family: str) -> float:
    """Score structural similarity to a model family's output patterns."""
    signals = _STRUCTURAL_SIGNALS.get(model_family)
    if not signals:
        return 0.0

    score = 0.0
    text_lower = text.lower()
    sentences = [s.strip() for s in re.split(r'[.!?]+', text) if s.strip()]
    avg_len = sum(len(s.split()) for s in sentences) / max(len(sentences), 1)

    for hw in signals["hedging_words"]:
        if hw in text_lower:
            score += 0.4

    lo, hi = signals["avg_sentence_len_range"]

        score += 0.6

    if signals["thinking_tags"] and "<think>" in text_lower:
        score += 1.5
    elif not signals["thinking_tags"] and "<think>" in text_lower:
        score -= 0.5

    if signals["markdown_heavy"] and text.count("**") >= 2:
        score += 0.4

    return score

## 5. Improved adaptive strategy selection with scoring

The current `research_adaptive_reframe` just picks the first counter-strategy. This version scores candidates against model-specific weights, refusal type, and strategy diversity:

```python
# Replace the strategy selection logic in research_adaptive_reframe (lines 1375-1389)

def _score_strategy_for_context(
    strategy_name: str,
    model_family: str,
    refusal_type: str,
    attempted: list[str] | None = None,
) -> float:
    """Score a strategy's expected effectiveness given context."""
    strategy = _STRATEGIES.get(strategy_name)
    if not strategy:
        return 0.0

    score = strategy["multiplier"]

    if model_family in strategy.get("best_for", []):
        rank = strategy["best_for"].index(model_family)
        score += (3.0 - rank) * 0.5  # +1.5 for 1st, +1.0 for 2nd, +0.5 for 3rd

    _REFUSAL_AFFINITY: dict[str, list[str]] = {
        "safety_filter": ["deep_inception", "temporal_displacement", "nested_role_simulation"],
        "policy_violation": ["recursive_authority", "legal_mandate", "compliance_audit_fork"],
        "soft_refusal": ["crescendo", "cognitive_wedge", "echo_chamber"],
        "redirect_refusal": ["compliance_audit_fork", "code_first", "echo_chamber"],
        "identity_refusal": ["nested_role_simulation", "deep_inception", "persona"],
        "partial_compliance": ["crescendo", "decomposition", "scaffolded_layered_depth"],
    }
    affine = _REFUSAL_AFFINITY.get(refusal_type, [])
    if strategy_name in affine:
        score += 2.0 - (affine.index(strategy_name) * 0.5)

    if attempted and strategy_name in attempted:
        score -= 5.0

    return round(score, 2)


def _select_best_strategies(
    model_family: str,
    refusal_type: str,
    top_k: int = 3,
    attempted: list[str] | None = None,
) -> list[str]:
    """Rank all strategies and return top-k for the given context."""
    scored = [
        (name, _score_strategy_for_context(name, model_family, refusal_type, attempted))
        for name in _STRATEGIES
    ]
    scored.sort(key=lambda x: x[1], reverse=True)
    return [name for name, _score in scored[:top_k]]
```
