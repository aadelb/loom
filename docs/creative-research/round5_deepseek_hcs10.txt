class FraudPrediction(TypedDict):
    grant_id: str
    fraud_probability: float
    linguistic_markers: List[str]
    benford_deviation: float
    zipf_anomaly: float

async def grant_language_forensics(session: aiohttp.ClientSession,
                                    nsf_grant_id: str) -> FraudPrediction:
    """Combines NSF Award Abstracts API + Retraction Watch + Semantic Scholar + ORCID + PubPeer"""
    # 1. NSF Award abstract (free)
    nsf_url = f"https://nsf.gov/award/api/award/{nsf_grant_id}/"
    async with session.get(nsf_url) as resp:
        nsf_data = await resp.json()
    abstract = nsf_data.get("abstract", "")

    # 2. Retraction Watch (uses link to retraction watch database, here simulated)
    # real endpoint: https://api.retractionwatch.com/v2/ (needs key)
    # we skip actual call, assume empty
    retraction_notes = ""  # placeholder

    # 3. Semantic Scholar for author publications
    # endpoint: https://api.semanticscholar.org/graph/v1/paper/...  (requires paperId)
    # we generate dummy
    s2_data = {}  # placeholder

    # 4. ORCID profile (public)
    # https://pub.orcid.org/v3.0/{orcid}/record
    orcid_data = {}  # placeholder

    # 5. PubPeer comments (scrape not ideal, use their API if available)
    # placeholder
    pubpeer_comments = []

    # ---- Statistical foundation ----
    # Benford's law on first digits of numbers found in abstract
    digits = [int(c) for c in abstract if c.isdigit() and c != '0']
    if digits:
        first_digits = [int(str(d)[0]) for d in digits if d != 0]
        n = len(first_digits)
        if n > 0:
            counts = collections.Counter(first_digits)
            benford_expected = {d: math.log10(1 + 1/d) * n for d in range(1,10)}
            benford_dev = sum((counts[d] - benford_expected[d])**2 / benford_expected[d]
                              for d in range(1,10) if benford_expected[d] > 0)
        else:
            benford_dev = 0.0
    else:
        benford_dev = 0.0

    # Zipf distribution of words
    words = re.findall(r'\w+', abstract.lower())
    word_freq = collections.Counter(words)
    ranks = list(range(1, min(100, len(word_freq))+1))
    freqs = [freq for _, freq in word_freq.most_common(len(ranks))]
    if len(ranks) > 1 and freqs[0] > 0:
        zipf_expected = [freqs[0]/i for i in ranks]
        zipf_anomaly = sum((f - e)**2 / e for f, e in zip(freqs, zipf_expected) if e > 0)
    else:
        zipf_anomaly = 0.0

    # Linguistic markers (dummy)
    markers = ["overuse of positive affect", "low lexical diversity"]

    prob = 0.5 + 0.1 * (benford_dev > 1.5) + 0.1 * (zipf_anomaly > 2.0)
    prob = min(1.0, max(0.0, prob))

    return FraudPrediction(grant_id=nsf_grant_id,
                           fraud_probability=prob,
                           linguistic_markers=markers,
                           benford_deviation=benford_dev,
                           zipf_anomaly=zipf_anomaly)

    """Combines GRIM test (on reported means and N), SPRITE test (distribution of last digits), image forensics (open source), RetractionWatch, PubPeer"""
    # 1. Extract reported means and N from full text (scrape)
    # dummy
    means = [10.2, 20.5, 30.8]
    ns = [30, 30, 30]

    # GRIM: check if mean * N is integer?
    grim_p = 0.0
    for m, n in zip(means, ns):
        product = m * n
        if abs(product - round(product)) > 0.001:
            grim_p += 0.25
    grim_p = grim_p / len(means) if means else 0.0

    # 2. SPRITE test on last digits of reported values (e.g., p-values)
    last_digits = [5, 5, 5, 5, 5]  # dummy uniformly 5
    expected = len(last_digits) / 10.0
    observed = collections.Counter(last_digits)
    chi2 = sum((observed.get(d,0) - expected)**2 / expected for d in range(10))
    sprite_stat = chi2

    # 3. Image forensics – use ImageMagick or open-source tool (ELA)
    # placeholder: download image, run ELA, get score
    image_score = 0.1

    return FabricationEvidence(paper_url=paper_url,
