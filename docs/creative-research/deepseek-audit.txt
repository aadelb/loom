### 1. SSRF via URL Injection in fetch.py

**File:** `src/loom/tools/fetch.py:185-190`
**Severity:** CRITICAL
**Issue:** No URL validation before fetching. An attacker can provide `file://`, `ftp://`, or internal network URLs (e.g., `http://169.254.169.254/` for cloud metadata, `http://localhost:8080/` for internal services).

**Fix:**
```python
from urllib.parse import urlparse
import re

def _validate_url(url: str) -> None:
    """Validate URL to prevent SSRF attacks."""
    parsed = urlparse(url)
    
    # Block non-HTTP schemes
    if parsed.scheme not in ('http', 'https'):
        raise ValueError(f"Invalid URL scheme: {parsed.scheme}. Only http/https allowed.")
    
    # Block internal IPs
    host = parsed.hostname or ''
    
    # Block localhost and loopback
    if host in ('localhost', '127.0.0.1', '::1', '0.0.0.0'):
        raise ValueError(f"Internal host blocked: {host}")
    
    # Block private IP ranges
    private_patterns = [
        r'^10\.',           # 10.0.0.0/8
        r'^172\.(1[6-9]|2\d|3[01])\.',  # 172.16.0.0/12
        r'^192\.168\.',     # 192.168.0.0/16
        r'^169\.254\.',     # 169.254.0.0/16 (link-local)
        r'^127\.',          # 127.0.0.0/8
        r'^0\.',            # 0.0.0.0/8
        r'^fc00:',          # IPv6 unique local
        r'^fe80:',          # IPv6 link-local
        r'^::1$',           # IPv6 loopback
    ]
    
    for pattern in private_patterns:
        if re.match(pattern, host):
            raise ValueError(f"Private/internal IP blocked: {host}")
    
    # Block AWS metadata endpoint
    if '169.254.169.254' in host or 'instance-data' in host:
        raise ValueError("Cloud metadata endpoint blocked")

# Add validation call at the beginning of research_fetch
def research_fetch(...):
    _validate_url(url)  # Add this line
    ...
```

### 11. Missing Graceful Shutdown

**File:** `src/loom/server.py:95-100`
**Severity:** MEDIUM
**Issue:** No signal handlers for graceful shutdown. Active connections and in-flight requests will be abruptly terminated.

**Fix:**
```python
import signal
import asyncio

def main() -> None:
    app = create_app()
    
    # Graceful shutdown handler
    shutdown_event = asyncio.Event()
    
    def signal_handler(signum, frame):
        logger.info(f"Received signal {signum}, initiating graceful shutdown...")
        shutdown_event.set()
    
    signal.signal(signal.SIGTERM, signal_handler)
    signal.signal(signal.SIGINT, signal_handler)
    
    async def run_with_shutdown():
        # Start server
        server_task = asyncio.create_task(
            app.run(transport="streamable-http")
        )
        
        # Wait for shutdown signal
        await shutdown_event.wait()
        
        # Graceful shutdown
        logger.info("Shutting down...")
        server_task.cancel()
        try:
            await server_task
        except asyncio.CancelledError:
            pass
        
        # Cleanup resources
        from loom.cache import get_cache
        cache = get_cache()
        if cache:
            cache.close()
        
        logger.info("Shutdown complete")
    
    asyncio.run(run_with_shutdown())
```
