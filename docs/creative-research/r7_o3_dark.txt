1. MULTI-ARCHIVE CONTENT RECOVERY
────────────────────────────────────────────────────────
Function signature  
async def recover_archives(url: str, limit: int = 100) -> ArchiveResult

Free data sources (examples)  
• Wayback CDX – https://web.archive.org/cdx/search/cdx  
• Memento TimeMap – http://memgator.cs.odu.edu/timemap/json  
• Common Crawl – http://index.commoncrawl.org/CC-MAIN-2023-50-index  
• Archive.today – https://archive.org/wayback/available  
• GitHub raw – https://raw.githubusercontent.com/… (for deleted repos)  
(+ 10 more can be appended exactly the same way)

```python
# archive_recover.py
import asyncio, json, httpx
from typing import List, TypedDict

class Snapshot(TypedDict):
    source: str
    timestamp: str
    original: str
    status: int

class ArchiveResult(TypedDict):
    target: str
    snapshots: List[Snapshot]

CDX  = "https://web.archive.org/cdx/search/cdx?url={}&output=json&fl=timestamp,original,statuscode&filter=statuscode:200"
MGT  = "http://memgator.cs.odu.edu/timemap/json/{}"
CC   = "http://index.commoncrawl.org/CC-MAIN-2023-50-index?url={}&output=json"
ARCH = "https://archive.org/wayback/available?url={}"
SRC  = {"CDX": CDX, "Memento": MGT, "CommonCrawl": CC, "ArchiveOrg": ARCH}

async def _fetch(client: httpx.AsyncClient, src: str, url: str):
    try:
        r = await client.get(SRC[src].format(url), timeout=20)
        r.raise_for_status()
        return src, r.text
    except Exception:
        return src, ""

async def recover_archives(url: str, limit: int = 100) -> ArchiveResult:
    async with httpx.AsyncClient(follow_redirects=True) as c:
        tasks = [asyncio.create_task(_fetch(c, s, url)) for s in SRC]
        raw = await asyncio.gather(*tasks)
    snaps: List[Snapshot] = []
    for src, body in raw:
        if not body:
            continue
        if src == "CDX":
            for ts, orig, sc in json.loads(body)[1:limit]:
                snaps.append({"source": src, "timestamp": ts, "original": orig, "status": int(sc)})
        elif src == "Memento":
            for m in json.loads(body).get("mementos", {}).get("list", [])[:limit]:
                snaps.append({"source": src, "timestamp": m["datetime"], "original": m["uri"], "status": 200})
        elif src == "CommonCrawl":
            for line in body.splitlines()[:limit]:
                j = json.loads(line)
                snaps.append({"source": src, "timestamp": j["timestamp"], "original": j["url"], "status": 200})
        elif src == "ArchiveOrg":
            for item in json.loads(body).get("archived_snapshots", {}).values():
                snaps.append({"source": src, "timestamp": item["timestamp"], "original": item["url"], "status": 200})
    return {"target": url, "snapshots": snaps}
```

2. HIDDEN-SERVICE DISCOVERY
────────────────────────────────────────────────────────
async def discover_onions(keyword: str, limit: int = 50) -> HiddenServiceResult

Data sources  
• Ahmia JSON ‑ https://ahmia.fi/api/v1/hidden_services/?query=  
• FreshOnion RSS – https://freshonions.org/feed/  
• Phobos onion mirrors – https://onion.live/api/onions  
• CT-Logs for `.onion` – https://crt.sh/?q=%25.onion&output=json  
• “DarknetStats” dump – https://raw.githubusercontent.com/…/darknetstats.json

```python
# discover_onions.py
import asyncio, re, json, httpx, feedparser
from typing import List, TypedDict, Set

class Onion(TypedDict):
    address: str
    source: str
    title: str

class HiddenServiceResult(TypedDict):
    keyword: str
    onions: List[Onion]

ENDPOINTS = {
    "ahmia": "https://ahmia.fi/api/v1/hidden_services/?query={}",
    "onionlive": "https://onion.live/api/onions?search={}",
    "crtsh": "https://crt.sh/?q={}&output=json",
}

async def _get(client: httpx.AsyncClient, name: str, q: str):
    try:
        r = await client.get(ENDPOINTS[name].format(q), timeout=20)
        return name, r.text
    except Exception:
        return name, ""

async def discover_onions(keyword: str, limit: int = 50) -> HiddenServiceResult:
    onions: Set[str] = set()
    results: List[Onion] = []
    async with httpx.AsyncClient() as c:
        tasks = [asyncio.create_task(_get(c, n, keyword)) for n in ENDPOINTS]
        tasks.append(asyncio.create_task(c.get("https://freshonions.org/feed/")))
        body = await asyncio.gather(*tasks)

    for name, data in body[:-1]:
        if not data:
            continue
        if name == "ahmia":
            for entry in json.loads(data)[:limit]:
                addr = entry["hidden_service"]
                if addr not in onions:
                    onions.add(addr)
                    results.append({"address": addr, "source": name, "title": entry.get("title", "")})
        elif name == "onionlive":
            for j in json.loads(data)["data"][:limit]:
                addr = j["address"]
                if addr not in onions:
                    onions.add(addr)
                    results.append({"address": addr, "source": name, "title": j.get("title", "")})
        elif name == "crtsh":
            for j in json.loads(data)[:limit]:
                m = re.findall(r"([a-z2-7]{16}\.onion)", j.get("name_value", ""))
                for addr in m:
                    if addr not in onions:
                        onions.add(addr)
                        results.append({"address": addr, "source": name, "title": "cert"})
    # FreshOnion RSS
    rss = feedparser.parse(body[-1].text)
    for e in rss.entries[:limit]:
        m = re.findall(r"[a-z2-7]{16}\.onion", e.summary)
        for addr in m:
            if addr not in onions:
                onions.add(addr)
                results.append({"address": addr, "source": "freshonion", "title": e.title})
    return {"keyword": keyword, "onions": results[:limit]}
```

────────────────────────────────────────────────────────
async def scan_exposure(email_or_domain: str) -> ExposureResult

• Shodan REST – https://api.shodan.io/shodan/host/search?query=  
• Google Dorks via SerpAPI free tier – https://serpapi.com/search.json?q=  
• AWS/GCP bucket check – https://storage.googleapis.com/{bucket}

```python
# data_exposure.py
import asyncio, os, json, httpx
from typing import List, TypedDict

class Leak(TypedDict):
    source: str
    reference: str
    detail: str

class ExposureResult(TypedDict):
    query: str
    leaks: List[Leak]

HIBP = "https://haveibeenpwned.com/api/v3/breachedaccount/{}"
GH   = "https://api.github.com/search/code?q={}&per_page=10"
PASTE= "https://psbdmp.ws/api/search/{}"
SHOD = "https://api.shodan.io/shodan/host/search?query={}&key={}"
SERP = "https://serpapi.com/search.json?q={}&api_key={}"

async def _req(c: httpx.AsyncClient, url: str, hdr=None):
    try:
        r = await c.get(url, headers=hdr or {}, timeout=20)
        return r.url.host, r.text
    except Exception:
        return "", ""

async def scan_exposure(email_or_domain: str) -> ExposureResult:
    k = email_or_domain
    leaks: List[Leak] = []
    hibp_hdr = {"hibp-api-key": os.getenv("HIBP_KEY", "")}
    async with httpx.AsyncClient() as c:
        urls = [
            HIBP.format(k),
            GH.format(k),
            PASTE.format(k),
            SHOD.format(k, os.getenv("SHODAN_KEY", "FREE")),
            SERP.format(f'inurl:"{k}"', os.getenv("SERP_KEY", "FREE")),
        ]
        tasks = [asyncio.create_task(_req(c, u, hibp_hdr if "haveibeenpwned" in u else None)) for u in urls]
        done = await asyncio.gather(*tasks)

    for host, body in done:
        if not body:
            continue
        if "haveibeenpwned" in host and body.startswith('['):
            for b in json.loads(body):
                leaks.append({"source": "HIBP", "reference": b["Name"], "detail": str(b["PwnCount"])})
        elif "github" in host:
            j = json.loads(body)
            for item in j.get("items", []):
                leaks.append({"source": "GitHub", "reference": item["html_url"], "detail": item["repository"]["full_name"]})
        elif "psbdmp" in host:
            for p in json.loads(body)[:10]:
                leaks.append({"source": "Pastebin", "reference": p["id"], "detail": p["title"]})
        elif "shodan" in host:
            for m in json.loads(body).get("matches", []):
                leaks.append({"source": "Shodan", "reference": m.get("ip_str"), "detail": m.get("hostnames")})
        elif "serpapi" in host:
            for r in json.loads(body).get("organic_results", [])[:10]:
                leaks.append({"source": "Google", "reference": r["link"], "detail": r["title"]})
    return {"query": k, "leaks": leaks}
```

async def aggregate_forums(keyword: str) -> ForumResult

```python
# dark_forum.py
import asyncio, json, re, httpx, html
from typing import List, TypedDict

class Post(TypedDict):
    source: str
    title: str
    url: str

class ForumResult(TypedDict):
    keyword: str
    posts: List[Post]

ENDP = {
    "darksearch": "https://darksearch.io/api/search?query={}",
    "intelx": "https://2.intelx.io/phonebook/search?k=free&q={}",
    "ahmia": "https://ahmia.fi/search/?q={}",
}

async def fetch(c: httpx.AsyncClient, name: str, kw: str):
    try:
        r = await c.get(ENDP[name].format(kw), timeout=20)
        return name, r.text
    except Exception:
        return name, ""

async def aggregate_forums(keyword: str) -> ForumResult:
    posts: List[Post] = []
    async with httpx.AsyncClient() as c:
        res = await asyncio.gather(*[fetch(c, n, keyword) for n in ENDP])
    for name, body in res:
        if not body:
            continue
        if name == "darksearch":
            for j in json.loads(body).get("data", [])[:20]:
                posts.append({"source": name, "title": j["title"], "url": j["link"]})
        elif name == "intelx":
            for j in json.loads(body).get("selectors", [])[:20]:
                posts.append({"source": name, "title": j.get("value",""), "url": j.get("selectorvalue","")})
        elif name == "ahmia":
            for m in re.finditer(r'<a href="(http[^"]+)"[^>]*>(.*?)</a>', body) :
                posts.append({"source": name, "title": html.unescape(m.group(2)), "url": m.group(1)})
    return {"keyword": keyword, "posts": posts}
```

5. INFRASTRUCTURE DE-ANONYMISATION
────────────────────────────────────────────────────────
async def deanonymize(host: str) -> InfraResult

Data sources  
• Censys universal search – https://search.censys.io/api/v2/hosts/search?q=  
• Favicon hash – https://raw.githubusercontent.com/devanshbatham/FavFreak/master/db/favfreak.json  
• crt.sh shared cert search – https://crt.sh/?q={}&output=json  
• JA3 dataset – https://ja3er.com/search/{md5}  
• Shodan host details – https://api.shodan.io/shodan/host/{ip}?key=…  

```python
# infra_deanon.py
import asyncio, json, hashlib, httpx
from typing import List, TypedDict

class Correlation(TypedDict):
    type: str
    value: str
    evidence: str

class InfraResult(TypedDict):
    host: str
    correlations: List[Correlation]

CRT = "https://crt.sh/?q={}&output=json"
CENS= "https://search.censys.io/api/v2/hosts/search?q={}"
SHO = "https://api.shodan.io/shodan/host/{}?key={}"

async def deanonymize(host: str) -> InfraResult:
    corr: List[Correlation] = []
    async with httpx.AsyncClient() as c:
        crt, cens = await asyncio.gather(
            c.get(CRT.format(host), timeout=20),
            c.get(CENS.format(host), timeout=20),
        )
    if crt.status_code==200:
        for j in json.loads(crt.text)[:30]:
            corr.append({"type":"cert","value":j["common_name"],"evidence":j["issuer_name"]})
    if cens.status_code==200:
        for h in cens.json().get("result", {}).get("hits", [])[:30]:
            ip=h["ip"]
            corr.append({"type":"censys","value":ip,"evidence":h["services"][0]["port"]})
    # quick favicon hash example (pull favicon, mmh3)
    try:
        fav = httpx.get(f"http://{host}/favicon.ico", timeout=5).content
        mmh = hashlib.md5(fav).hexdigest()
        corr.append({"type":"favicon","value":mmh,"evidence":"md5"})
    except Exception:
        pass
    return {"host": host, "correlations": corr}
```

```python
# btc_flow.py
import asyncio, json, httpx
from typing import List, Dict, TypedDict

class Hop(TypedDict):
    txid: str
    to: str
    amount: float

class BTCFlow(TypedDict):
    start: str
    hops: List[Hop]

RAW = "https://blockchain.info/rawaddr/{}"
STREAM = "https://blockstream.info/api/address/{}/txs"

async def trace_btc(address: str, depth: int = 1) -> BTCFlow:
    hops: List[Hop] = []
    async with httpx.AsyncClient() as c:
        r = await c.get(RAW.format(address), timeout=20)
    if r.status_code!=200:
        return {"start": address, "hops": hops}
    data = r.json()
    for tx in data.get("txs", [])[:depth]:
        txid = tx["hash"]
        out = tx["out"][0]
        hops.append({"txid": txid, "to": out["addr"], "amount": out["value"]/1e8})
    return {"start": address, "hops": hops}
```

```python
# stego_detector.py
import asyncio, io, re, json, httpx, struct
from typing import List, TypedDict

class Indicator(TypedDict):
    type: str
    detail: str

class StegoReport(TypedDict):
    url: str
    indicators: List[Indicator]

HOMO = "https://raw.githubusercontent.com/codebox/homoglyph/master/data.json"

def _lsb(img: bytes) -> bool:
    bits = "".join([str(b & 1) for b in img[:8000]])
    return "1010101111001101" in bits

async def detect_stego(file_url: str) -> StegoReport:
    ind: List[Indicator] = []
    async with httpx.AsyncClient() as c:
        img = (await c.get(file_url, timeout=20)).content
        homo = json.loads((await c.get(HOMO, timeout=20)).text)
    if _lsb(img):
        ind.append({"type":"LSB","detail":"Binary pattern found"})
    # EXIF quick
    if b"Exif" in img[:100]:
        ind.append({"type":"EXIF","detail":"EXIF header present"})
    # Unicode homoglyph demo
    suspect = re.findall(r'[\u0400-\u04FF]+', img.decode("utf8","ignore"))
    if suspect:
        ind.append({"type":"Unicode","detail":str(len(suspect))+" Cyrillic homographs"})
    return {"url": file_url, "indicators": ind}
```

```python
import asyncio, json, httpx, time

    value: str
```

```python
# vuln_fusion.py
import asyncio, csv, io, json, httpx
from typing import List, TypedDict

class Vuln(TypedDict):
    source: str
    id: str
    title: str

class VulnFusion(TypedDict):
    query: str
    vulns: List[Vuln]

NVD   = "https://services.nvd.nist.gov/rest/json/cves/2.0?keywordSearch={}"
ECSV  = "https://gitlab.com/exploit-database/exploitdb/-/raw/main/files_exploits.csv"
GHADV = "https://api.github.com/search/advisories?query={}"
CISA  = "https://www.cisa.gov/sites/default/files/feeds/known_exploited_vulnerabilities.json"
VULNS = "https://vulners.com/api/v3/search/lucene/?query={}"

async def vuln_fusion(cve_or_kw: str) -> VulnFusion:
    vulns: List[Vuln] = []
    async with httpx.AsyncClient() as c:
        nvd, csvd, gh, cisa, vul = await asyncio.gather(
            c.get(NVD.format(cve_or_kw), timeout=20),
            c.get(ECSV, timeout=20),
            c.get(GHADV.format(cve_or_kw), timeout=20),
            c.get(CISA, timeout=20),
            c.get(VULNS.format(cve_or_kw), timeout=20),
        )
    if nvd.status_code==200:
        for item in nvd.json().get("vulnerabilities", [])[:20]:
            cve=item["cve"]["id"]; title=item["cve"]["descriptions"][0]["value"]
            vulns.append({"source":"NVD","id":cve,"title":title})
    if csvd.status_code==200:
        buf=io.StringIO(csvd.text); reader=csv.reader(buf)
        for row in list(reader)[:200]:
            if cve_or_kw.lower() in row[2].lower():
                vulns.append({"source":"ExploitDB","id":row[0],"title":row[2]})
    if gh.status_code==200:
        for adv in gh.json().get("items", [])[:20]:
            vulns.append({"source":"GitHub","id":adv["ghsa_id"],"title":adv["summary"]})
    if cisa.status_code==200:
        for v in cisa.json()["vulnerabilities"][:100]:
            if cve_or_kw in v["cveID"]:
                vulns.append({"source":"CISA KEV","id":v["cveID"],"title":v["vendorProject"]})
    if vul.status_code==200:
        for d in vul.json()["data"].get("search", [])[:20]:
            vulns.append({"source":"Vulners","id":d["id"],"title":d["title"]})
    return {"query": cve_or_kw, "vulns": vulns}
```
