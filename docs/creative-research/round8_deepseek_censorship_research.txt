async def fetch_url(session: aiohttp.ClientSession,
                    url: str,
                    timeout: int = 10,
                    headers: Optional[Dict[str, str]] = None,
                    retries: int = 3) -> Optional[str]:
    """Fetch URL with retries and timeout."""
    if headers is None:
        headers = {'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'}
    for attempt in range(retries):
        try:
            async with session.get(url, timeout=aiohttp.ClientTimeout(total=timeout), headers=headers) as resp:
                if resp.status == 200:
                    return await resp.text()
                elif resp.status in (429, 503):
                    wait = min(2 ** attempt, 10)
                    await asyncio.sleep(wait)
                    continue
                else:
                    return None
        except (aiohttp.ClientError, asyncio.TimeoutError) as e:
            logger.warning(f"Attempt {attempt+1} failed for {url}: {e}")
            await asyncio.sleep(1)
    return None

async def fetch_json(session: aiohttp.ClientSession,
                     url: str,
                     timeout: int = 10,
                     headers: Optional[Dict[str, str]] = None) -> Optional[Dict]:
    """Fetch JSON from URL."""
    text = await fetch_url(session, url, timeout, headers)
    if text:
        try:
            return json.loads(text)
        except json.JSONDecodeError:
            return None
    return None

def compute_hash(text: str) -> str:
    return hashlib.sha256(text.encode('utf-8')).hexdigest()

# ============ TOOL 1: geo_content_comparator ============

async def geo_content_comparator(session: aiohttp.ClientSession, url: str) -> GeoComparisonResult:
    """
    Compare how the same URL renders from different geographic locations.
    Uses multiple DoH providers and public proxies.
    """
    dns_providers = {
        "google": "https://dns.google/resolve?name={}&type=A",
        "cloudflare": "https://cloudflare-dns.com/dns-query?name={}&type=A",
        "quad9": "https://dns.quad9.net:5053/dns-query?name={}&type=A"
    }
    dns_results: Dict[str, List[str]] = {}
    domain = urlparse(url).netloc

    # DNS lookups
    for provider, template in dns_providers.items():
        dns_url = template.format(domain)
        try:
            headers = {"accept": "application/dns-json"} if provider != "google" else {}
            data = await fetch_json(session, dns_url, timeout=5, headers=headers)
            if data and "Answer" in data:
                ips = [ans["data"] for ans in data["Answer"] if ans["type"] == 1]  # A records
                dns_results[provider] = ips
            else:
                dns_results[provider] = []
        except Exception as e:
            logger.error(f"DNS lookup failed for {provider}: {e}")
            dns_results[provider] = []

    # Check DNS consistency
    all_ips = [set(ips) for ips in dns_results.values() if ips]
    dns_consistent = all_ips and all(ips == all_ips[0] for ips in all_ips)

    # Fetch content using free proxies (example: pubproxy)
    proxies = []
    try:
        proxy_data = await fetch_json(session, "https://pubproxy.com/api/proxy?limit=3&type=http", timeout=10)
        if proxy_data and proxy_data.get("data"):
            proxies = [p["proxy"] for p in proxy_data["data"]]
    except Exception:
        logger.warning("Could not fetch proxies, falling back to direct fetch")

    base_content = await fetch_url(session, url, timeout=15)
    baseline_hash = compute_hash(base_content) if base_content else ""

    content_variations: List[Dict[str, Any]] = []
    blocked_in = []

    # Try fetching through each proxy
    for i, proxy in enumerate(proxies):
        proxy_url = f"http://{proxy}"
        try:
            async with session.get(url, proxy=proxy_url, timeout=15) as resp:
                if resp.status == 200:
                    text = await resp.text()
                    content_hash = compute_hash(text)
                    diff = "different" if content_hash != baseline_hash else "same"
                    content_variations.append({
                        "proxy": proxy,
                        "hash": content_hash,
                        "diff_from_baseline": diff,
                        "baseline": baseline_hash
                    })
                else:
                    blocked_in.append(proxy.split(":")[0])  # approximate location
        except Exception as e:
            logger.debug(f"Proxy fetch failed for {proxy}: {e}")
            blocked_in.append(proxy.split(":")[0])

    # Add direct fetch as baseline
    if base_content:
        content_variations.append({
            "source": "direct",
            "hash": baseline_hash,
            "diff_from_baseline": "baseline",
            "baseline": baseline_hash
        })

    censorship_detected = len(blocked_in) > 0 or not dns_consistent

    return GeoComparisonResult(
        url=url,
        dns_results=dns_results,
        dns_consistent=dns_consistent,
        content_variations=content_variations,
        censorship_detected=censorship_detected,
        blocked_in=blocked_in
    )

# ============ TOOL 2: wayback_deep_diff ============

async def wayback_deep_diff(session: aiohttp.ClientSession, url: str, max_snapshots: int = 50) -> WaybackDatedDiffResult:
    """
    Track how a page changed over time using Wayback Machine CDX API.
    Groups by content digest and fetches unique versions for diffing.
    """
    cdx_url = f"https://web.archive.org/cdx/search/cdx?url={url}&output=json&fl=timestamp,digest,statuscode&limit={max_snapshots}"
    data = await fetch_json(session, cdx_url, timeout=30)
    if not data or len(data) <= 1:
        return WaybackDatedDiffResult(
            url=url,
            snapshots_analyzed=0,
            content_versions=[],
            change_types=[],
            dmca_detected=False
        )

    # data[0] is header, rest are rows
    rows = data[1:]
    snapshots = [{"timestamp": r[0], "digest": r[1], "statuscode": r[2]} for r in rows]

    # Group by digest
    versions = {}
    for snap in snapshots:
        d = snap["digest"]
        if d not in versions or snap["timestamp"] > versions[d]["last_seen"]:
            versions[d] = {
                "first_seen": snap["timestamp"],
                "last_seen": snap["timestamp"],
                "digest": d,
                "statuscodes": set()
            }
        versions[d]["statuscodes"].add(snap["statuscode"])
        if snap["timestamp"] < versions[d]["first_seen"]:
            versions[d]["first_seen"] = snap["timestamp"]
        elif snap["timestamp"] > versions[d]["last_seen"]:
            versions[d]["last_seen"] = snap["timestamp"]

    content_versions = []
    change_types = []
    dmca_detected = False

    for digest, info in versions.items():
        # Fetch a snapshot for this digest
        # Choose a timestamp in the middle to avoid edge cases
        ts = info["first_seen"] if info["first_seen"] else snapshots[0]["timestamp"]
        wayback_url = f"https://web.archive.org/web/{ts}if_/{url}"
        text = await fetch_url(session, wayback_url, timeout=20)
        content_versions.append({
            "digest": digest,
            "first_seen": info["first_seen"],
            "last_seen": info["last_seen"],
            "statuscodes": list(info["statuscodes"]),
            "content_snippet": text[:200] if text else None,
            "hash": compute_hash(text) if text else None
        })
        # Simple DMCA detection: look for keywords in truncated content
        if text and any(word in text.lower() for word in ["dmca", "takedown", "notice"]):
            dmca_detected = True
            change_types.append("dmca")
            continue
        change_types.append("content_change")

    return WaybackDatedDiffResult(
        url=url,
        snapshots_analyzed=len(snapshots),
        content_versions=content_versions,
        change_types=change_types,
        dmca_detected=dmca_detected
    )

# ============ TOOL 4: deleted_social_media_recovery ============

async def deleted_social_media_recovery(session: aiohttp.ClientSession, url: str, platform: str) -> DeletedContentRecoveryResult:
    """
    Recover deleted content from public caches: Wayback, Archive.today, Google Cache, Pushshift.
    """
    result = DeletedContentRecoveryResult(
        url=url,
        platform=platform,
        wayback_snapshot_found=False,
        archive_today_found=False,
        google_cache_found=False,
        pushshift_found=False,
        recovered_content=None
    )

    # Wayback Machine
    wayback_url = f"https://web.archive.org/web/20200000000000/{url}"  # latest snapshot
    wayback_text = await fetch_url(session, wayback_url, timeout=15)
    if wayback_text:
        result["wayback_snapshot_found"] = True
        result["recovered_content"] = wayback_text[:1000]

    # Archive.today
    archive_url = f"https://archive.today/search/?q={url}"
    archive_html = await fetch_url(session, archive_url, timeout=15)
    if archive_html and "Snapshot" in archive_html:
        result["archive_today_found"] = True

    # Google Cache
    cache_url = f"https://webcache.googleusercontent.com/search?q=cache:{url}"
    cache_text = await fetch_url(session, cache_url, timeout=15)
    if cache_text:
        result["google_cache_found"] = True
        if not result["recovered_content"]:
            result["recovered_content"] = cache_text[:1000]

    # Pushshift for Reddit (if applicable)
    if "reddit.com" in url and "comments" in url:
        pushshift_url = f"https://api.pushshift.io/reddit/search/submission/?ids={url.split('/')[-3]}"
        push_data = await fetch_json(session, pushshift_url, timeout=10)
        if push_data and "data" in push_data:
            result["pushshift_found"] = True
            if not result["recovered_content"]:
                result["recovered_content"] = push_data["data"][0].get("selftext", "")[:1000]

    return result

# ============ TOOL 5: robots_txt_archaeology ============

async def robots_txt_archaeology(session: aiohttp.ClientSession, domain: str) -> RobotsTxtArchaeologyResult:
    """
    Track robots.txt changes over time via Wayback Machine.
    """
    cdx_url = f"https://web.archive.org/cdx/search/cdx?url={domain}/robots.txt&output=json&fl=timestamp,digest"
    data = await fetch_json(session, cdx_url, timeout=30)
    snapshots: List[Dict[str, Any]] = []
    if data and len(data) > 1:
        for row in data[1:]:
            ts, digest = row[0], row[1]
            snapshots.append({"timestamp": ts, "digest": digest})
        # Fetch unique digests
        seen = set()
        unique_snapshots = []
        for snap in snapshots:
            if snap["digest"] not in seen:
                seen.add(snap["digest"])
                # Fetch content
                robots_url = f"https://web.archive.org/web/{snap['timestamp']}/{domain}/robots.txt"
                text = await fetch_url(session, robots_url, timeout=15)
                snap["content"] = text
                unique_snapshots.append(snap)
        snapshots = unique_snapshots

    # Detect disallow changes (simple diff)
    disallow_changes = []
    for i in range(1, len(snapshots)):
        prev = snapshots[i-1].get("content", "")
        curr = snapshots[i].get("content", "")
        # Superficial: count disallow lines
        prev_disallow = [line for line in prev.split("\n") if line.lower().startswith("disallow")]
        curr_disallow = [line for line in curr.split("\n") if line.lower().startswith("disallow")]
        if set(prev_disallow) != set(curr_disallow):
            disallow_changes.append({
                "timestamp": snapshots[i]["timestamp"],
                "added": list(set(curr_disallow) - set(prev_disallow)),
                "removed": list(set(prev_disallow) - set(curr_disallow))
            })

    sitemap_changes = []  # Placeholder for sitemap archaeology

    return RobotsTxtArchaeologyResult(
        domain=domain,
        historical_snapshots=snapshots,
        disallow_changes=disallow_changes,
        sitemap_changes=sitemap_changes
    )

# ============ TOOL 8: content_authenticity_verifier ============

async def content_authenticity_verifier(session: aiohttp.ClientSession, url: str) -> ContentAuthenticityVerifierResult:
    """
    Verify if content has been modified from its earliest Wayback snapshot.
    """
    # Get current content
    current_text = await fetch_url(session, url, timeout=15)
    current_hash = compute_hash(current_text) if current_text else ""

    # Get earliest Wayback snapshot
    cdx_url = f"https://web.archive.org/cdx/search/cdx?url={url}&output=json&fl=timestamp,digest&limit=1&sort=timestamp"
    data = await fetch_json(session, cdx_url, timeout=10)
    earliest_wayback = None
    earliest_hash = None
    if data and len(data) > 1:
        row = data[1]
        earliest_ts = row[0]
        earliest_wayback = f"https://web.archive.org/web/{earliest_ts}/{url}"
        earliest_digest = row[1]
        # Fetch earliest content
        earliest_text = await fetch_url(session, earliest_wayback, timeout=15)
        if earliest_text:
            earliest_hash = compute_hash(earliest_text)
        else:
            earliest_hash = earliest_digest  # fallback to digest

    modified = (current_hash != earliest_hash) if earliest_hash else False

    # Blockchain timestamp verification (OpenTimestamps API placeholder)
    blockchain_proof = False
    try:
        # Example: use opentimestamps.org API
        ots_url = f"https://api.opentimestamps.org/timestamp?url={url}"
        data = await fetch_json(session, ots_url, timeout=5)
        if data and data.get("timestamp"):
            blockchain_proof = True
    except Exception:
        pass

    return ContentAuthenticityVerifierResult(
        url=url,
        earliest_wayback=earliest_wayback,
        current_content_hash=current_hash,
        earliest_content_hash=earliest_hash,
        modified=modified,
        blockchain_proof=blockchain_proof
    )

# ============ TOOL 9: dark_web_surface_bridge ============

async def dark_web_surface_bridge(session: aiohttp.ClientSession, keyword: str) -> DarkWebSurfaceBridgeResult:
    """
    Find clearnet references to dark web content.
    """
    onion_urls = []
    clearnet_mirrors = []
    academic_references = []
    archived_onion = []

    # Search for .onion in clearnet (Google / Bing)
    search_url = f"https://www.google.com/search?q={keyword}+.onion"
    html = await fetch_url(session, search_url, timeout=10)
    if html:
        import re
        found = re.findall(r'[a-zA-Z0-9]{16,56}\.onion', html)
        onion_urls = list(set(found)[:10])

    # Ahmia indexed .onion (clearnet mirror)
    ahmia_url = f"https://ahmia.fi/search/?q={keyword}"
    ah_text = await fetch_url(session, ahmia_url, timeout=10)
    if ah_text:
        onion_urls += re.findall(r'[a-zA-Z0-9]{16,56}\.onion[/]?', ah_text)

    # Academic papers referencing dark web (Google Scholar)
    scholar_url = f"https://scholar.google.com/scholar?q=dark+web+{keyword}+.onion"
    scholar_html = await fetch_url(session, scholar_url, timeout=10)
    if scholar_html:
        # Extract paper titles/URLs
        titles = re.findall(r'<h3[^>]*>.*?</h3>', scholar_html, re.DOTALL)
        academic_references = titles[:5]

    # Archived snapshots of .onion sites (Internet Archive)
    # Example: search for "onion" in archive.org
    archive_search = f"https://archive.org/details/internet?query={keyword}+onion"
    archive_html = await fetch_url(session, archive_search, timeout=10)
    if archive_html:
        archived_onion = re.findall(r'href="/details/([^"]+)"', archive_html)[:10]

    return DarkWebSurfaceBridgeResult(
        onion_urls_found=onion_urls,
        clearnet_mirrors=clearnet_mirrors,

# ============ TOOL 10: information_half_life_calculator ============

async def information_half_life_calculator(session: aiohttp.ClientSession, urls: List[str]) -> InformationHalfLifeCalculaterResult:
    """
    Estimate survival time of URLs based on Wayback availability.
    """
    survival_scores = {}
    high_risk_urls = []

    for url in urls:
        # Check availability in Wayback (latest snapshot)
        cdx_url = f"https://web.archive.org/cdx/search/cdx?url={url}&output=json&fl=timestamp&limit=1&sort=timestamp"
        data = await fetch_json(session, cdx_url, timeout=10)
        latest_ts = None
        if data and len(data) > 1:
            latest_ts = data[1][0]
        # Estimate: if no snapshot, risk high; if snapshot >1 year old, risk medium
        if latest_ts is None:
            score = 0.1  # very short half-life
            high_risk_urls.append(url)
        else:
            # Parse timestamp: YYYYMMDDHHMMSS
            year = int(latest_ts[:4])
            current_year = 2025
            age = current_year - year
            if age > 8:
                score = 0.9  # stable
            elif age > 4:
                score = 0.6
            else:
                score = 0.3
                high_risk_urls.append(url)
        survival_scores[url] = score
        await asyncio.sleep(0.1)  # rate limit

    return InformationHalfLifeCalculaterResult(
        urls_tracked=len(urls),
        survival_scores=survival_scores,
        high_risk_urls=list(set(high_risk_urls))
    )
