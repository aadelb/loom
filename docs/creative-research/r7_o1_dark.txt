1) ARCHIVE RECOVERY (Wayback, Common Crawl, Memento, Deleted Packages)
--------------------------------------------------------------------------------
1  from typing import TypedDict, List
2  import aiohttp
3  
4  class ArchiveRecoveryResult(TypedDict):
5      url: str
6      wayback_snapshots: List[str]
7      commoncrawl_snapshots: List[str]
8      memento_snapshots: List[str]
9      deleted_packages: List[str]
10 
11 async def archive_recovery(url: str) -> ArchiveRecoveryResult:
12     """
13     Searches Wayback Machine, Common Crawl, Memento, and references
14     for deleted packages to uncover historical data of a given URL.
15     """
16     wayback = []
17     ccrawl = []
18     memento = []
19     deleted = []
20     async with aiohttp.ClientSession() as session:
21         # Query Wayback
22         wb_api = "http://archive.org/wayback/available"
23         async with session.get(wb_api, params={"url": url}) as r:
24             if r.status == 200:
25                 data = await r.json()
26                 if data.get("archived_snapshots"):
27                     wayback.append(data["archived_snapshots"]["closest"]["url"])
28         # PLACEHOLDER: Common Crawl, Memento, deleted packages lookups
29     return {
30         "url": url, "wayback_snapshots": wayback, "commoncrawl_snapshots": ccrawl,
         "memento_snapshots": memento, "deleted_packages": deleted
}
