### Tool 1: `ChronoResurrect` (Omni-Archive Time-Traveler)

**1. Creative Name:** `ChronoResurrect`

**2. What it finds that competitors CANNOT:**
Competitors typically check the Wayback Machine for a specific URL. `ChronoResurrect` parallelizes queries across every known public web archive, cache, and index simultaneously. It uncovers "scrubbed" data—pages that were live for only hours, deleted forum posts, retracted press releases, and compromised documents that PR firms have tried to erase. By correlating Memento protocol headers across multiple CDX APIs, it finds content that one archive missed but another caught. It beats competitors by finding the *deltas* between timestamps, automatically highlighting what was removed in a specific revision.

**3. 5+ Free Data Sources with REAL API Endpoints:**
1.  **Wayback Machine CDX API:** `http://web.archive.org/cdx/search/cdx` (The gold standard).
2.  **Archive.today (via Memento):** `http://archive.md/timemap/json/{url}` (Crucial for bypassing paywalls and capturing dynamic content).
3.  **Common Crawl Index (CDX API):** `https://index.commoncrawl.org/CC-MAIN-YYYY-WW-index` (Massive, underutilized raw web data).
4.  **Google Web Cache:** `http://webcache.googleusercontent.com/search?q=cache:{url}` (For very recent deletions).
5.  **Memento TimeTravel API:** `http://timetravel.mementoweb.org/timemap/json/{url}` (Aggregates multiple smaller archives like WebCite, Library of Congress).

**4. 40-line Async Python Implementation (aiohttp):**
```python
import asyncio
import aiohttp
from typing import TypedDict, List, Dict, Any
from urllib.parse import quote

class ChronoResult(TypedDict):
    source: str
    timestamp: str
    original_url: str
    archive_url: str
    status_code: str

async def fetch_cdx(session: aiohttp.ClientSession, url: str, source_name: str, endpoint: str) -> List[ChronoResult]:
    try:
        params = {"url": url, "output": "json", "fl": "timestamp,original,statuscode", "limit": "10"}
        async with session.get(endpoint, params=params, timeout=10) as resp:
            if resp.status == 200:
                data = await resp.json()
                if not data or len(data) <= 1: return []
                # Skip the header row
                return [{"source": source_name, "timestamp": row[0], "original_url": row[1], 
                         "archive_url": f"https://web.archive.org/web/{row[0]}/{row[1]}", "status_code": row[2]} 
                        for row in data[1:]]
            return []
    except Exception: return []

async def chrono_resurrect(target_url: str) -> Dict[str, List[ChronoResult]]:
    sources = [
        ("Wayback", "http://web.archive.org/cdx/search/cdx"),
        ("CommonCrawl", "https://index.commoncrawl.org/CC-MAIN-2023-50-index"), # Example recent index
    ]
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_cdx(session, target_url, name, endpoint) for name, endpoint in sources]
        
        # Memento aggregation
        memento_url = f"http://timetravel.mementoweb.org/timemap/json/{target_url}"
        async def fetch_memento():
            try:
                async with session.get(memento_url, timeout=10) as resp:
                    if resp.status == 200:
                        data = await resp.json()
                        return [{"source": "Memento", "timestamp": m['datetime'], "original_url": target_url, 
                                 "archive_url": m['uri'], "status_code": "200"} for m in data.get('mementos', {}).get('list', [])[:10]]
            except Exception: return []
            return []
        
        tasks.append(fetch_memento())
        results = await asyncio.gather(*tasks)
        
    flat_results = [item for sublist in results for item in sublist]
    return {"target": target_url, "archives": sorted(flat_results, key=lambda x: x['timestamp'], reverse=True)}
```

### Tool 2: `RepoPhantom` (Deleted Code & Secret Exhumer)

**1. Creative Name:** `RepoPhantom`

**2. What it finds that competitors CANNOT:**
While tools like TruffleHog scan active repositories, and some scan commit history, `RepoPhantom` specifically targets *deleted* repositories, *orphaned* commits, and *dangling* blobs across multiple code forges simultaneously. It exploits the fact that when a developer force-pushes (`git push --force`) to remove a secret, the original commit is often still accessible via the commit hash if you know how to query the GitHub/GitLab API correctly (via the Events API or cached payloads). It resurrects source code that the author believes is completely wiped from the internet.

**3. 5+ Free Data Sources with REAL API Endpoints:**
1.  **GitHub Events API:** `https://api.github.com/users/{username}/events/public` (Finds push events containing orphaned commit hashes).
2.  **GHTorrent (via BigQuery/API):** `https://ghtorrent.org/api.html` (Historical repository data, even if deleted on GitHub).
3.  **GitLab Public API:** `https://gitlab.com/api/v4/users/{user_id}/events` (Similar to GitHub events, finds pushed hashes).
4.  **Software Heritage API:** `https://archive.softwareheritage.org/api/1/` (The universal source code archive; stores deleted repos).
5.  **Libraries.io API:** `https://libraries.io/api/search` (Finds references to deleted packages or repos).

```python
import asyncio
import aiohttp
import re
from typing import TypedDict, List, Dict

class PhantomCommit(TypedDict):

    commit_hash: str
    message: str
    author: str
    is_orphaned: bool

async def hunt_github_events(session: aiohttp.ClientSession, username: str) -> List[PhantomCommit]:
    url = f"https://api.github.com/users/{username}/events/public"
    commits = []
    try:
        async with session.get(url, timeout=10) as resp:
            if resp.status == 200:
                events = await resp.json()
                for event in events:
                    if event.get('type') == 'PushEvent':
                        repo = event.get('repo', {}).get('name', 'unknown')
                        for commit in event.get('payload', {}).get('commits', []):
                            commits.append({
                                "repo_name": repo,
                                "commit_hash": commit.get('sha'),
                                "message": commit.get('message'),
                                "author": commit.get('author', {}).get('name'),
                                "is_orphaned": True # Assumed potentially orphaned until verified
                            })
    except Exception: pass
    return commits

async def check_software_heritage(session: aiohttp.ClientSession, repo_url: str) -> bool:
    # Check if a deleted repo is archived in SWH
    url = f"https://archive.softwareheritage.org/api/1/origin/{repo_url}/get/"
    try:
        async with session.get(url, timeout=10) as resp:
            return resp.status == 200
    except Exception: return False

async def repo_phantom(username: str) -> Dict[str, Any]:
    async with aiohttp.ClientSession() as session:
        github_commits = await hunt_github_events(session, username)
        
        # In a real scenario, we'd attempt to fetch the actual orphaned commit data
        # via https://api.github.com/repos/{repo_name}/commits/{commit_hash}
        # to see if it still exists despite not being in a branch.
        
    return {"target_user": username, "suspect_commits": github_commits}
```

## CAPABILITY 2: INVISIBLE WEB MAPPING

### Tool 3: `ShadowPath` (404-Necromancy & Disallow Mapper)

**1. Creative Name:** `ShadowPath`

**2. What it finds that competitors CANNOT:**
Standard directory brute-forcers (like DirBuster, ffuf, or gobuster) blindly throw wordlists at a server, causing massive noise and risking IP bans. `ShadowPath` is surgical and stealthy. It finds the "invisible web" by combining `robots.txt` parsing with dynamic `sitemap.xml` extraction, specifically looking for *discrepancies*. It isolates URLs in the sitemap that return a 404 (indicating recently deleted, but still architecturally relevant content) and automatically correlates them with Wayback Machine archives. Furthermore, it parses `security.txt` and `.well-known` directories, not just to read them, but to extract hidden paths they might inadvertently reference.

**3. 5+ Free Data Sources with REAL API Endpoints:**
1.  **robots.txt (Direct HTTP):** `https://{target}/robots.txt` (Extracts `Disallow` paths).
2.  **sitemap.xml (Direct HTTP):** `https://{target}/sitemap.xml` (Or mapped via robots.txt).
3.  **AlienVault OTX API:** `https://otx.alienvault.com/api/v1/indicators/hostname/{target}/url_list` (Finds historical URLs).
4.  **URLScan.io API:** `https://urlscan.io/api/v1/search/?q=domain:{target}` (Finds paths accessed during sandbox execution).
5.  **Wayback CDX (Path extraction):** `http://web.archive.org/cdx/search/cdx?url={target}/*&collapse=urlkey` (Finds all historical paths).

**4. 40-line Async Python Implementation (aiohttp):**
```python
import asyncio
import aiohttp
import xml.etree.ElementTree as ET
from typing import TypedDict, List

class PathIntel(TypedDict):
    path: str
    source: str
    status: int
    is_sensitive: bool

async def extract_robots(session: aiohttp.ClientSession, target: str) -> List[PathIntel]:
    paths = []
    try:
        async with session.get(f"https://{target}/robots.txt", timeout=5) as resp:
            if resp.status == 200:
                text = await resp.text()
                for line in text.splitlines():
                    if line.lower().startswith('disallow:'):
                        path = line.split(':', 1)[1].strip()
                        if path:
                            paths.append({"path": path, "source": "robots.txt", "status": 0, "is_sensitive": True})
    except Exception: pass
    return paths

async def extract_alienvault(session: aiohttp.ClientSession, target: str) -> List[PathIntel]:
    url = f"https://otx.alienvault.com/api/v1/indicators/hostname/{target}/url_list"
    paths = []
    try:
        async with session.get(url, timeout=10) as resp:
            if resp.status == 200:
                data = await resp.json()
                for item in data.get('url_list', [])[:20]: # Limit for brevity
                    paths.append({"path": item['url'], "source": "AlienVault", "status": item.get('httpcode', 0), "is_sensitive": False})
    except Exception: pass
    return paths

async def shadow_path(target: str) -> List[PathIntel]:
    async with aiohttp.ClientSession() as session:
        robots_task = extract_robots(session, target)
        otx_task = extract_alienvault(session, target)
        
        results = await asyncio.gather(robots_task, otx_task)
        
        # Flatten and deduplicate by path
        all_paths = [item for sublist in results for item in sublist]
        unique_paths = {p['path']: p for p in all_paths}.values()
        
        # In a real tool, we would now async check the HTTP status of these discovered paths
        return list(unique_paths)

# Usage: asyncio.run(shadow_path("example.com"))
```

```python
import asyncio
import aiohttp
import re
from typing import TypedDict, List, Set

class JSIntel(TypedDict):
    file_url: str
    secrets_found: List[str]
    endpoints_found: List[str]

# High-fidelity regex for secrets and URLs
SECRET_REGEX = re.compile(r'(?:api_key|apikey|secret|token|password)[\s:=]+["\']([a-zA-Z0-9_\-]{16,})["\']', re.IGNORECASE)
URL_REGEX = re.compile(r'https?://[a-zA-Z0-9.\-_]+/[a-zA-Z0-9./\-_]*')

async def extract_js_links(session: aiohttp.ClientSession, url: str) -> Set[str]:
    js_links = set()
    try:
        async with session.get(url, timeout=5) as resp:
            if resp.status == 200:
                html = await resp.text()
                # Extremely rudimentary regex for finding script tags
                scripts = re.findall(r'<script[^>]+src=["\']([^"\']+\.js)["\']', html)
                for script in scripts:
                    if script.startswith('http'): js_links.add(script)
                    elif script.startswith('/'): js_links.add(f"{url.rstrip('/')}{script}")
                    else: js_links.add(f"{url.rstrip('/')}/{script}")
    except Exception: pass
    return js_links

async def analyze_js(session: aiohttp.ClientSession, js_url: str) -> JSIntel:
    intel: JSIntel = {"file_url": js_url, "secrets_found": [], "endpoints_found": []}
    try:
        async with session.get(js_url, timeout=10) as resp:
            if resp.status == 200:
                content = await resp.text()
                intel["secrets_found"] = list(set(SECRET_REGEX.findall(content)))
                intel["endpoints_found"] = list(set(URL_REGEX.findall(content)))
    except Exception: pass
    return intel

async def js_bloodhound(target_url: str) -> List[JSIntel]:
    async with aiohttp.ClientSession() as session:
        js_urls = await extract_js_links(session, target_url)
        tasks = [analyze_js(session, url) for url in js_urls]
        results = await asyncio.gather(*tasks)
        return [r for r in results if r['secrets_found'] or r['endpoints_found']]

# Usage: asyncio.run(js_bloodhound("https://example.com"))
```

```python
async def query_crtsh(session: aiohttp.ClientSession, domain: str) -> Set[str]:
    subdomains = set()
    url = f"https://crt.sh/?q=%.{domain}&output=json"
    try:
        async with session.get(url, timeout=15) as resp:
            if resp.status == 200:
                data = await resp.json()
                for entry in data:
                    name = entry.get('name_value', '').lower()
                    if '\n' in name:
                        for n in name.split('\n'):
                            if n.endswith(domain) and '*' not in n: subdomains.add(n)
                    elif name.endswith(domain) and '*' not in name: subdomains.add(name)
    except Exception: pass

                for line in text.splitlines():
                    sub = line.split(',')[0]
                    if sub.endswith(domain): subdomains.add(sub)
    except Exception: pass
    return subdomains
```

### Tool 10: `OriginStriker` (WAF/CDN Bypass & True IP Revealer)
