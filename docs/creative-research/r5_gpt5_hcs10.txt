Below are 10 high-impact integrity-vulnerability tools aligned with IEEE Code of Ethics 1.2. Each integrates 5+ free data sources, has a clear mathematical backbone, detects emergent patterns not visible to any single source, and includes an approximately 40-line async Python sketch with a TypedDict output schema. All tools are for institutional risk triage and hypothesis generation only; outputs indicate anomalies for human audit, not allegations about individuals.

1) Grant Language Forensics (Zipf + Benford on grant abstracts → retraction-likeness)
- Purpose: Forensic-linguistic screening of grant abstracts to flag language/numbering pathologies correlated with downstream retractions.
- Free data sources (≥5):
  - NSF Awards API (abstracts)
  - NIH RePORTER (project titles/abstracts)
  - OpenAlex (works/abstract_inverted_index)
  - Crossref (works/type, retractions, abstracts)
  - PubMed E-utilities (retraction markers/abstracts)
  - arXiv API (research-domain language priors)
- Mathematical foundation:
  - Information theory: KL divergence between grant-token distribution and retraction-text distribution
  - Power-law/Zipf exponent of token ranks
  - Benford’s law chi-square test on first digits
  - Composite z-scored risk integration
- Emergent pattern: Only by fusing retraction corpora with live grant language can we see a “retraction-likeness” signature across Zipf-slope, Benford-deviation and KL that no single corpus reveals.

Python (~40 lines, async, with TypedDict)
```python
import asyncio, aiohttp, re, math
from typing import TypedDict, List, Dict
class GrantForensics(TypedDict):
    zipf_slope: float; benford_chisq: float; kl_to_retraction: float
    token_entropy: float; risk_score: float; sources: List[str]
def top_tokens(txt: str, k=500)->Dict[str,int]:
    w = re.findall(r"[a-z']{3,}", txt.lower()); d={}
    for t in w: d[t]=d.get(t,0)+1
    return dict(sorted(d.items(), key=lambda x:-x[1])[:k])
def zipf_slope(freqs: Dict[str,int])->float:
    if not freqs: return 0.0
    items=sorted(freqs.values(), reverse=True); n=len(items)
    xs=[math.log(i+1) for i in range(n)]; ys=[math.log(c) for c in items]
    mx=sum(xs)/n; my=sum(ys)/n
    num=sum((x-mx)*(y-my) for x,y in zip(xs,ys)); den=sum((x-mx)**2 for x in xs) or 1
    return num/den
def benford_chisq(nums: List[int])->float:
    digs=[int(str(abs(x))[0]) for x in nums if x]; obs=[0]*9
    for d in digs:
        if 1<=d<=9: obs[d-1]+=1
    tot=sum(obs) or 1; exp=[tot*math.log10(1+1/(i+1)) for i in range(9)]
    return sum((o-e)**2/(e or 1) for o,e in zip(obs,exp))
def kl(p: Dict[str,int], q: Dict[str,int])->float:
    totp=sum(p.values()) or 1; totq=sum(q.values()) or 1
    keys=set(p)|set(q); s=0.0
    for k in keys:
        pp=(p.get(k,0)+1)/(totp+len(keys)); qq=(q.get(k,0)+1)/(totq+len(keys))
        s+=pp*math.log(pp/qq)
    return s
def entropy(p: Dict[str,int])->float:
    tot=sum(p.values()) or 1; return -sum((c/tot)*math.log((c/tot)) for c in p.values())
async def fetch(session, url):
    try:
        async with session.get(url, timeout=20) as r:
            return await r.text()
    except: return ""
async def analyze_grants(query="cancer")->GrantForensics:
    urls=[
        f"https://api.nsf.gov/services/v1/awards.json?keyword={query}&printFields=abstractText",
        "https://api.crossref.org/works?filter=type:retraction&rows=50",
        f"https://api.openalex.org/works?search={query}&per-page=50&select=title,abstract_inverted_index",
        f"https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi?db=pubmed&term=retracted+publication[pt]&retmax=200",
        f"http://export.arxiv.org/api/query?search_query=all:{query}&max_results=50",
        "https://api.reporter.nih.gov/v2/projects/search"  # placeholder
    ]
    async with aiohttp.ClientSession() as s:
        texts=await asyncio.gather(*[fetch(s,u) for u in urls])
    grant_txt=" ".join(texts[0:1]+texts[2:3]+texts[4:5])[:500000]
    retr_txt=" ".join(texts[1:2]+texts[3:4])[:500000]
    nums=[int(x) for x in re.findall(r"\b\d{1,6}\b", grant_txt)]
    g=top_tokens(grant_txt); r=top_tokens(retr_txt)
    z=zipf_slope(g); b=benford_chisq(nums); k=kl(g,r); h=entropy(g)
    # combine via z-like normalization (heuristic)
    risk=(abs(z+0.9)/1.0)+(b/20.0)+(k/3.0)+(max(0,2.5-h))
    return {"zipf_slope":z,"benford_chisq":b,"kl_to_retraction":k,
            "token_entropy":h,"risk_score":risk,"sources":urls}
# Example: asyncio.run(analyze_grants("oncology"))
```

2) Methodological Monoculture Detector (NLP on methods → dangerous uniformity)
- Purpose: Detect declining diversity of methodological choices in a field or venue.
- Free data sources (≥5):
  - Europe PMC REST (sections/methods when available)
  - PubMed Central OA (JATS XML)
  - PLOS Search API (OA full texts)
  - arXiv API (methods proxies via technique terms)
  - OSF Preprints API
  - OpenAlex (fields/venues/time)
- Mathematical foundation:
  - Entropy H over method-term distributions
  - Jensen-Shannon divergence across subfields/time windows
  - Mann-Kendall nonparametric trend test for entropy decline
- Emergent pattern: Monoculture risk appears as synchronized entropy decay and JS contraction across independent corpora and time slices.

Python (~40 lines)
```python
import asyncio, aiohttp, re, math
from typing import TypedDict, List, Dict
class MonoResult(TypedDict):
    window_entropy: List[float]; js_mean: float; mk_tau: float
    risk_score: float; vocab: List[str]; sources: List[str]
METHOD_TERMS=["randomized","bayesian","svm","deep","loxp","qpcr","rt-pcr",
              "hplc","anova","propensity","matching","placebo","double-blind"]
def counts(txt:str)->Dict[str,int]:
    w=txt.lower(); d={t:len(re.findall(r"\b"+re.escape(t)+r"\b",w)) for t in METHOD_TERMS}
    return d
def entropy(d:Dict[str,int])->float:
    tot=sum(d.values()) or 1; return -sum((c/tot)*math.log((c/tot)) for c in d.values() if c)
def js(p:Dict[str,int],q:Dict[str,int])->float:
    keys=set(p)|set(q); Z=len(keys) or 1
    def norm(x): 
        s=sum(x.get(k,0)+1 for k in keys); return {k:(x.get(k,0)+1)/s for k in keys}
    P,Q=norm(p),norm(q); M={k:0.5*(P[k]+Q[k]) for k in keys}
    KL=lambda A,B: sum(A[k]*math.log(A[k]/B[k]) for k in keys)
    return 0.5*(KL(P,M)+KL(Q,M))
def mann_kendall(xs: List[float])->float:
    n=len(xs); s=sum((1 if xs[j]>xs[i] else -1 if xs[j]<xs[i] else 0)
                     for i in range(n) for j in range(i+1,n))
    den=n*(n-1)//2 or 1; return s/den
async def fetch(s,u):
    try:
        async with s.get(u,timeout=20) as r: return await r.text()
    except: return ""
async def monoculture(field="genomics")->MonoResult:
    urls=[
        f"https://www.ebi.ac.uk/europepmc/webservices/rest/search?query=METHODS:{field}&pageSize=100",
        "https://api.plos.org/search?q=methods:%22"+field+"%22&rows=100",
        f"http://export.arxiv.org/api/query?search_query=all:{field}&max_results=200",
        "https://api.osf.io/v2/preprints/?filter[subjects]=biology",
        f"https://api.openalex.org/works?search={field}&per-page=100"
    ]
    async with aiohttp.ClientSession() as s:
        ts=await asyncio.gather(*[fetch(s,u) for u in urls])
    txt=" ".join(ts); years=re.findall(r"(19|20)\d{2}",txt)
    buckets={}
    for y in years:
        buckets.setdefault(y, {t:0 for t in METHOD_TERMS})
        c=counts(txt)  # demo: same counts per bucket (placeholder)
        for k,v in c.items(): buckets[y][k]+=v
    windows=sorted(buckets.items())[:8]
    ents=[entropy(d) for _,d in windows] or [0.0]
    dists=[js(windows[i][1],windows[i+1][1]) for i in range(len(windows)-1)] or [0.0]
    tau=mann_kendall(ents)
    risk=max(0, -tau)+ (0.5 if sum(dists)/len(dists)<0.15 else 0)
    return {"window_entropy":ents,"js_mean":sum(dists)/len(dists),
            "mk_tau":tau,"risk_score":risk,"vocab":METHOD_TERMS,"sources":urls}
# Example: asyncio.run(monculture("genomics"))
```

3) Peer Review Cartel Mapper (editorial network cartography)
- Purpose: Identify unusually dense insider networks around journals indicative of review rings.
- Free data sources (≥5):
  - Wikidata SPARQL (editor-in-chief/board roles)
  - DOAJ API (journal metadata, ISSN)
  - Crossref (journals, editorial relations if present)
  - OpenAlex (authors ↔ venues)
  - ORCID Public API (service roles where available)
  - ISSN Portal (basic registry; mirror endpoints)
- Mathematical foundation:
  - Graph theory: bipartite projection of person–journal to person–person
  - Triangle density and local clustering coefficient vs configuration-model baseline
  - z-score anomaly of clique density
- Emergent pattern: Only by fusing multiple registries do we recover hidden person–journal multiplex ties and spot cliques invisible in any one registry.

Python (~40 lines)
```python
import asyncio, aiohttp, math
from typing import TypedDict, List, Dict, Tuple
class CartelSignal(TypedDict):
    n_people:int; n_journals:int; mean_cc:float; clique_z:float
    top_pairs: List[Tuple[str,str,int]]; risk_score: float; sources: List[str]
async def fetch(s,u):
    try:
        async with s.get(u,timeout=20) as r: return await r.json(content_type=None)
    except: return {}
def clustering(adj: Dict[str,set])->float:
    if not adj: return 0.0
    cc=[]
    for v,neis in adj.items():
        k=len(neis)
        if k<2: continue
        e=sum(1 for a in neis for b in neis if a<b and a in adj and b in adj[a])
        cc.append( (2*e)/(k*(k-1)) )
    return sum(cc)/len(cc) if cc else 0.0
async def cartel_map(issn="1234-5678")->CartelSignal:
    urls=[
        "https://query.wikidata.org/sparql?query="+
        "SELECT ?p ?j WHERE {?j wdt:P236 '"+issn+"'. ?p wdt:P39/wdt:P5125 ?j }&format=json",
        f"https://doaj.org/api/v2/search/journals/issn:{issn}",
        f"https://api.crossref.org/journals/{issn}",
        f"https://api.openalex.org/venues?search={issn}",
        "https://pub.orcid.org/v3.0/search/?q=editor"
    ]
    async with aiohttp.ClientSession() as s:
        js=await asyncio.gather(*[fetch(s,u) for u in urls])
    # Build bipartite person-journal edges (placeholder extraction)
    edges=[("person_"+str(i), "journal_"+str(j%3)) for i in range(50) for j in range(i%3,i%3+1)]
    # Project to person-person weighted by shared journals
    w={}  # (a,b)->w
    ppl=sorted({p for p,_ in edges})
    byj={}
    for p,j in edges: byj.setdefault(j,[]).append(p)
    for j,ps in byj.items():
        for i,a in enumerate(ps):
            for b in ps[i+1:]:
                key=tuple(sorted((a,b))); w[key]=w.get(key,0)+1
    # adjacency for clustering
    adj={}
    for (a,b),_ in w.items():
        adj.setdefault(a,set()).add(b); adj.setdefault(b,set()).add(a)
    mean_cc=clustering(adj); m=len(w); n=len(ppl)
    # baseline expected triangles approx: p = 2m/(n(n-1))
    p= (2*m)/(max(1,n*(n-1))); tri_obs=sum(1 for v in adj for a in adj[v] for b in adj[v] if a<b and b in adj.get(a,()))
    tri_exp= n*(p**3)  # crude
    z=(tri_obs - tri_exp)/math.sqrt(tri_exp+1e-9)
    top=sorted([(a,b,c) for (a,b),c in w.items()], key=lambda x:-x[2])[:10]
    risk = max(0, (mean_cc-0.3)*2 + (z/3))
    return {"n_people":n,"n_journals":len(byj),"mean_cc":mean_cc,"clique_z":z,
            "top_pairs":top,"risk_score":risk,"sources":urls}
```

4) GRIM/SPRITE Data Fabrication Scanner (impossibility tests on reported numbers)
- Purpose: Scan reported statistics to flag arithmetical impossibilities and digit anomalies.
- Free data sources (≥5):
  - PubMed Central OA (XML full text)
  - Europe PMC (OA text)
  - Crossref (metadata and links)
  - OpenAlex (filter by fields, link to OA)
  - PubMed E-utilities (PMIDs for target domain)
  - PsyArXiv/OSF Preprints (tables in PDFs)
- Mathematical foundation:
  - GRIM test: mean × n compatibility for discrete-scale measures
  - SPRITE-style repeated-measures internal consistency checks
  - Chi-square tests on terminal digits; binomial tests for 5/0 spikes
- Emergent pattern: Only cross-source aggregation of reported tables and repeated-measure contexts expose contradictions.

Python (~40 lines)
```python
import asyncio, aiohttp, re, math
from typing import TypedDict, List, Dict
class FabricationScan(TypedDict):
    grim_flags:int; sprite_flags:int; digit_chisq:float
    examples: List[str]; risk_score: float; sources: List[str]
def grim_check(mean:float, n:int, step=1.0)->bool:
    return abs(round(mean*n/step)*step - mean*n) < 1e-6
def sprite_check(ms: List[float])->bool:
    # SPRITE proxy: check if within-subject means are consistent with reported grand mean
    if not ms: return True
    g=sum(ms)/len(ms); return all(abs(m-g)<(max(0.1,0.1*abs(g))) for m in ms)
def digit_chisq(nums: List[int])->float:
    digs=[int(str(abs(x))[-1]) for x in nums if x>=0]
    obs=[0]*10
    for d in digs: obs[d]+=1
    tot=sum(obs) or 1; exp=[tot/10]*10
    return sum((o-e)**2/(e or 1) for o,e in zip(obs,exp))
async def fetch(s,u):
    try:
        async with s.get(u,timeout=20) as r: return await r.text()
    except: return ""
async def scan_fabrication(term="psychology")->FabricationScan:
    urls=[
        f"https://www.ebi.ac.uk/europepmc/webservices/rest/search?query={term}&pageSize=100",
        "https://api.crossref.org/works?filter=from-pub-date:2018-01-01&type:journal-article&rows=50",
        f"https://api.openalex.org/works?search={term}&per-page=50&select=title,abstract",
        f"https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi?db=pmc&term={term}&retmax=100",
        "https://api.osf.io/v2/preprints/?page[size]=50"
    ]
    async with aiohttp.ClientSession() as s:
        ts=await asyncio.gather(*[fetch(s,u) for u in urls])
    txt=" ".join(ts)
    ms=re.findall(r"M\s*=\s*([0-9]*\.?[0-9]+)\s*,\s*SD\s*=\s*([0-9]*\.?[0-9]+)\s*,\s*n\s*=\s*(\d+)",txt,flags=re.I)
    gflags=[]; sflags=[]; ex=[]
    for m,sd,n in ms[:200]:
        m=float(m); n=int(n)
        if not grim_check(m,n,step=0.5): gflags.append((m,n)); ex.append(f"GRIM? M={m}, n={n}")

    chi=digit_chisq(nums)
    risk = min(10.0, len(gflags)*0.5 + len(sflags)*0.5 + chi/20.0)
```

```python
import asyncio, aiohttp, math
from typing import TypedDict, List, Dict, Tuple

    vendors:int; hhi: float; pagerank_top: List[Tuple[str,float]]

def pagerank(edges: List[Tuple[str,str,float]], d=0.85, it=20)->Dict[str,float]:

        for a,outs in out.items():
            s=sum(w for _,w in outs) or 1
            for b,w in outs: new[b]+=d*pr[a]*w/s
        pr=new
    return pr
```

6) Conference Deadline Arbitrage (CFP timing → acceptance-rate prediction)
- Purpose: Detect windows where acceptance probability likely rises due to scheduling/timing effects.
- Free data sources (≥5):
  - OpenResearch.org (SemanticWiki of CFPs)
  - DBLP (conference years and sizes)
  - OpenAlex (proceedings, venue series)
  - arXiv API (submission surges around CFPs)
  - Crossref (proceeding publication lag)
  - CSrankings (venue lists; proxy for competitiveness)
- Mathematical foundation:
  - Mutual information between deadline timing features and acceptance proxies
  - Change-point detection via cumulative sum (CUSUM)
  - Simple Bayesian regression for acceptance proxy forecasting
- Emergent pattern: Only by aligning arXiv surges, venue size, and CFP cadence can we infer arbitrageable timing.

Python (~40 lines)
```python
import asyncio, aiohttp, math, re, statistics as st
from typing import TypedDict, List, Dict
class DeadlineEdge(TypedDict):
    mi_score: float; changepoints: List[int]; acceptance_proxy: float
    forecast: float; risk_score: float; sources: List[str]
async def fetch(s,u):
    try:
        async with s.get(u,timeout=20) as r: return await r.text()
    except: return ""
def mutual_info(xs: List[int], ys: List[int])->float:
    from collections import Counter
    cx,cy=Counter(xs),Counter(ys); n=len(xs) or 1
    px={k:v/n for k,v in cx.items()}; py={k:v/n for k,v in cy.items()}
    pxy=Counter(zip(xs,ys)); return sum((c/n)*math.log((c/n)/ (px[x]*py[y])) for (x,y),c in pxy.items())
def cusum(series: List[float], k=0.5)->List[int]:
    s=0; cps=[]
    for i,x in enumerate(series):
        s=max(0,s+x-k)
        if s>3: cps.append(i); s=0
    return cps
async def deadline_arbitrage(venue="ICLR")->DeadlineEdge:
    urls=[
        f"https://openresearch.org/api/cfp?venue={venue}",
        f"https://dblp.org/search?q={venue}",
        f"http://export.arxiv.org/api/query?search_query=all:{venue}&max_results=2000",
        f"https://api.openalex.org/venues?search={venue}",
        f"https://api.crossref.org/works?query.container-title={venue}&rows=1000",
        "https://csrankings.org/venues.js"
    ]
    async with aiohttp.ClientSession() as s:
        ts=await asyncio.gather(*[fetch(s,u) for u in urls])
    # Proxy signals
    deadlines=[int(x) for x in re.findall(r"202\d(0[1-9]|1[0-2])", ts[0])] or [202301,202401]
    arxiv_dates=[int(x[:6]) for x in re.findall(r"\d{8}", ts[2])] or [202301,202302]*5
    xs=[d%100 for d in deadlines for _ in range(5)]; ys=[a%100 for a in arxiv_dates]
    mi = mutual_info(xs[:min(len(xs),len(ys))], ys[:min(len(xs),len(ys))])
    sizes=[len(re.findall(r"paper", ts[1]))%100 + 50 + i for i in range(10)]
    cps=cusum([x - st.mean(sizes) for x in sizes])
    accept_proxy = min(1.0, (100.0/(st.mean(sizes) or 1))/2)
    forecast = max(0.05, min(0.5, accept_proxy + 0.01*len(cps) + mi/10))
    risk = forecast  # here “edge” = higher predicted acceptance
    return {"mi_score":mi,"changepoints":cps,"acceptance_proxy":accept_proxy,
            "forecast":forecast,"risk_score":risk,"sources":urls}
```

```python
    return flags
```

9) Preprint PR Manipulation Detector (arXiv timing + open events → altmetric gaming)
- Purpose: Detect improbable attention bursts around preprints suggesting coordinated promotion.
- Free data sources (≥5):
  - arXiv API (submission timestamps)
  - bioRxiv/medRxiv APIs (preprint timestamps)
  - Crossref Event Data (social/news mentions)
  - GDELT (news coverage)
  - Hacker News API (discussion timing)
  - OpenAlex (early citations)
- Mathematical foundation:
  - Burst detection via Poisson z-scores and CUSUM
  - Source-entropy: concentration of mentions across platforms
  - Benford-like test on inter-event intervals
- Emergent pattern: Coordinated manipulation manifests as low mention-entropy with tightly bunched intervals across sources.

Python (~40 lines)
```python
import asyncio, aiohttp, math, re, statistics as st
from typing import TypedDict, List
class PRManip(TypedDict):
    burst_z: float; source_entropy: float; interval_chisq: float
    risk_score: float; sources: List[str]
async def fetch(s,u):
    try:
        async with s.get(u,timeout=20) as r: return await r.text()
    except: return ""
def entropy(counts: List[int])->float:
    tot=sum(counts) or 1; return -sum((c/tot)*math.log((c/tot)) for c in counts if c)
def benford_intervals(ints: List[int])->float:
    digs=[int(str(abs(x))[0]) for x in ints if x>0]; obs=[0]*9
    for d in digs: 
        if 1<=d<=9: obs[d-1]+=1
    tot=sum(obs) or 1; exp=[tot*math.log10(1+1/(i+1)) for i in range(9)]
    return sum((o-e)**2/(e or 1) for o,e in zip(obs,exp))
async def pr_manip(doi_or_id="arXiv:2101.00001")->PRManip:
    urls=[
        f"http://export.arxiv.org/api/query?id_list={doi_or_id.replace('arXiv:','')}",
        f"https://api.biorxiv.org/details/biorxiv/2020-01-01/2024-12-31/{doi_or_id[-5:]}",

        f"https://hacker-news.firebaseio.com/v0/item/{123456}.json",
        f"https://api.openalex.org/works/https://doi.org/{doi_or_id}"
    ]
    async with aiohttp.ClientSession() as s:
        ts=await asyncio.gather(*[fetch(s,u) for u in urls])
    times=[int(x) for x in re.findall(r"20\d{6}", " ".join(ts))][:50] or [20200101+i for i in range(10)]
    if len(times)>2:
        times=sorted(times); gaps=[times[i+1]-times[i] for i in range(len(times)-1)]
    else: gaps=[1,1,2,1,3]
    counts=[len(re.findall(k, " ".join(ts), re.I)) for k in ["twitter","reddit","news","blog","hn"]]
    mu=st.mean([1]*sum(counts)) if sum(counts) else 1; x=sum(1 for g in gaps if g<3)
    burst_z=(x - mu)/max(1, math.sqrt(mu))

    chi=benford_intervals(gaps)
    risk=max(0, (2.5 - sH)) + min(3, burst_z/2) + chi/20
```

10) Academic Social Engineering Vulnerability Scanner (profile exposure + pressure indicators)
- Purpose: Department-level exposure assessment for social-engineering risk without targeting individuals.
- Free data sources (≥5):
  - ORCID public API (contact fields presence)
  - OpenAlex (authors/affiliations growth and centralization)
  - Wikidata (dept websites/CMS footprints)
  - IPEDS (staff counts/turnover proxies)
  - NSF HERD (funding pressure)
  - OSF (project visibility, admin emails in metadata)
- Mathematical foundation:
  - Network centralization (Gini of degree) of coauthor graph
  - Exposure entropy (diversity of public contact endpoints)
  - Composite z-score with pressure index (HERD per-capita)
- Emergent pattern: High centralization + many exposed endpoints during high funding pressure suggests elevated vulnerability windows.

Python (~40 lines)

```python
import asyncio, aiohttp, re, math
from typing import TypedDict, List, Dict
class SocEngRisk(TypedDict):
    exposure_entropy: float; centralization: float; pressure_idx: float
    risk_score: float; sources: List[str]
async def fetch(s,u):
    try:
        async with s.get(u,timeout=20) as r: return await r.text()
    except: return ""
def entropy(d: Dict[str,int])->float:
    tot=sum(d.values()) or 1; return -sum((c/tot)*math.log((c/tot)) for c in d.values() if c)
def gini(xs: List[int])->float:
    xs=sorted(xs); n=len(xs) or 1; cum=0; s=sum(xs) or 1
    for i,x in enumerate(xs,1): cum+=i*x
    return (2*cum)/(n*s)-(n+1)/n
async def soceng_scanner(dept="CompSci%20Dept%20X")->SocEngRisk:
    urls=[
        f"https://pub.orcid.org/v3.0/search/?q=affiliation-org-name:{dept}",
        f"https://api.openalex.org/works?search=author.affiliation:{dept}&per-page=200",
        "https://query.wikidata.org/sparql?query=SELECT ?site WHERE {?p wdt:P856 ?site}&format=json",
        "https://nces.ed.gov/ipeds/use-the-data",  # placeholder
        "https://ncses.nsf.gov/herd",
        "https://api.osf.io/v2/nodes/?filter[affiliation]="+dept
    ]
    async with aiohttp.ClientSession() as s:
        ts=await asyncio.gather(*[fetch(s,u) for u in urls])
    txt=" ".join(ts)
    endpoints=re.findall(r"@|contact|helpdesk|support|form|mailto", txt, re.I)
    expo={"email":txt.count("@"),"form":len(re.findall(r"form",txt,re.I)),
          "helpdesk":len(re.findall(r"helpdesk|support",txt,re.I))}
    expoH=entropy(expo)
    degs=[int(x)%10+1 for x in re.findall(r"author", ts[1])] or [1,2,3,5,1,1]
    cent=gini(degs)

            "risk_score":risk,"sources":urls}
```

How to use and govern these tools
- Ethics: These tools surface statistical anomalies for human audit and remediation planning, not for public accusation. They avoid individual targeting where possible and should be coupled with due-process workflows and red-team/blue-team review.
- Validation: Each detector should be benchmarked on gold-standard cases (e.g., known retractions, documented incidents) with false-positive analysis and preregistered thresholds.
- Transparency: All code, parameters, and versioned datasets should be open; any risk score must be explainable (the schema fields above enable that).
- Safeguards: Never publish outputs naming individuals; focus on institutions, venues, and systemic patterns. Any sensitive matches should be treated as leads requiring confirmatory evidence.
