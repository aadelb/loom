# Utility – async HTTP session with retries
# ================================================================
class AsyncSession:
    """Wrapper around aiohttp.ClientSession with retry and rate limiting."""
    def __init__(self, max_conn=10, total_retries=3):
        self.semaphore = asyncio.Semaphore(max_conn)
        self.session = None
        self.total_retries = total_retries

    async def __aenter__(self):
        self.session = aiohttp.ClientSession(
            headers={'User-Agent': Config.USER_AGENT},
            timeout=aiohttp.ClientTimeout(total=30)
        )
        return self

    async def __aexit__(self, *args):
        await self.session.close()

    async def request(self, method, url, **kwargs) -> aiohttp.ClientResponse:
        async with self.semaphore:
            for attempt in range(self.total_retries):
                try:
                    async with self.session.request(method, url, **kwargs) as resp:
                        if resp.status == 429:
                            wait = int(resp.headers.get('Retry-After', 10))
                            logger.warning(f"Rate limited on {url}, waiting {wait}s")
                            await asyncio.sleep(wait)
                            continue
                        await resp.read()
                        return resp
                except (aiohttp.ClientError, asyncio.TimeoutError) as e:
                    logger.warning(f"Request failed ({attempt+1}/{self.total_retries}): {e}")
                    await asyncio.sleep(2 ** attempt)
            raise RuntimeError(f"Failed after {self.total_retries} retries: {url}")

class OnionEnumerator:
    """
    Discovers .onion services using multiple public sources.
    """
    async def discover(self, keyword: str = "") -> List[OnionService]:
        services = []

        async with AsyncSession() as session:
            # Ahmia
            ahmia_task = self._ahmia_search(session, keyword)
            # Fresh Onion RSS (mock)

            darkfail_task = self._darkfail_mirrors(session)
            # Certificate Transparency (crt.sh) for .onion
            crtsh_task = self._crtsh_onion(session)

            results = await asyncio.gather(ahmia_task, rss_task, darkfail_task, crtsh_task,
                                           return_exceptions=True)

        for r in results:
            if isinstance(r, list):
                services.extend(r)

        # Deduplicate
        seen = set()
        unique = []
        for s in services:
            if s['onion_url'] not in seen:
                seen.add(s['onion_url'])
                unique.append(s)
        return unique

    async def _ahmia_search(self, session, keyword):
        """Ahmia search API."""
        if not keyword:
            return []
        url = f"https://ahmia.fi/search/?q={quote(keyword)}"
        try:
            resp = await session.request('GET', url)
            html = await resp.text()
            # Extract .onion links
            onions = re.findall(r'[a-z2-7]{56}\.onion', html)
            return [{'onion_url': f"http://{on}/", 'discovered_by': 'ahmia', 'title': None, 'category': None} for on in set(onions)]
        except Exception as e:
            logger.error(f"Ahmia search failed: {e}")
            return []

    async def _fresh_onion_rss(self, session):
        """Parse Fresh Onion RSS feed (example: http://zqktlwi4fecvo6ri.onion/wiki/)."""
        # For demonstration we use a clearnet mirror
        rss_url = "https://raw.githubusercontent.com/alecmuffett/onion-sites/master/sites.csv"
        try:
            resp = await session.request('GET', rss_url)
            text = await resp.text()
            lines = text.split('\n')[1:]  # skip header
            services = []
            for line in lines:
                parts = line.split(',')
                if len(parts) >= 1:
                    onion = parts[0].strip()
                    if onion.endswith('.onion'):
                        services.append({'onion_url': f"http://{onion}/", 'discovered_by': 'fresh_onion_rss', 'title': None, 'category': None})
            return services
        except Exception as e:
            logger.error(f"Fresh Onion RSS failed: {e}")
            return []

    async def _darkfail_mirrors(self, session):
        """Mirror list from Dark.fail."""
        url = "https://dark.fail/api/mirrors.json"
        try:
            resp = await session.request('GET', url)
            data = await resp.json()
            services = []
            for item in data:
                if 'onion' in item.get('url', ''):
                    services.append({'onion_url': item['url'], 'discovered_by': 'darkfail', 'title': item.get('name', ''), 'category': item.get('category', '')})

    async def _crtsh_onion(self, session):

        except Exception as e:
            logger.error(f"crt.sh failed: {e}")
            return []

            items = data.get('items', [])

class StegDetector:
    """Detect hidden data in images, text, DNS, etc."""
    async def analyze_image(self, image_path: str) -> Dict:
        if not Image:
            return {'error': 'Pillow not installed'}
        try:
            img = Image.open(image_path)
            pixels = img.load()
            width, height = img.size
            # LSB extraction (rudimentary)
            binary = ''
            for y in range(height):
                for x in range(width):
                    r, g, b = pixels[x, y][:3]
                    binary += str(r & 1) + str(g & 1) + str(b & 1)
                    if len(binary) > 80000:  # arbitrary limit
                        break
                if len(binary) > 80000:
                    break
            # Convert binary to ASCII
            message = ''
            for i in range(0, len(binary), 8):
                byte = binary[i:i+8]
                if len(byte) < 8:
                    break
                char = chr(int(byte, 2))
                if char.isprintable() or char in '\n\r\t':
                    message += char
            return {
                'file': image_path,
                'lsb_message': message[:100],
                'exif': img.info if hasattr(img, 'info') else {}
            }
        except Exception as e:
            return {'error': str(e)}

    async def analyze_whitespace(self, text: str) -> str:
        """Detect whitespace steganography (binary encoding in spaces/tabs)."""
        lines = text.split('\n')
        extracted = []
        for line in lines:
            if line.endswith(' ') or line.endswith('\t'):
                # simple decode: space=0, tab=1
                bits = ''.join('1' if c == '\t' else '0' for c in line if c in ' \t')
                for i in range(0, len(bits), 8):
                    byte = bits[i:i+8]
                    if len(byte) == 8:
                        extracted.append(chr(int(byte, 2)))
        return ''.join(extracted)

            for doc in data2.get('response', {}).get('docs', []):
                results.append({'source': 'archive.org', 'title': doc.get('title', ''), 'url': doc.get('identifier', '')})
        return results

# ================================================================
class ThreatProfiler:
    """Build profiles from OSINT clues (stylometry, timezone, PGP)."""
    async def profile(self, username: str) -> Dict:
        async with AsyncSession() as session:
            # Search for username across platforms (simplified sherlock)
            platforms = {
                'reddit': f"https://www.reddit.com/user/{username}/about.json",
                'github': f"https://api.github.com/users/{username}",
                'twitter': f"https://api.twitter.com/2/users/by/username/{username}"  # needs bearer
            }
            results = {}
            for platform, url in platforms.items():
                try:
                    resp = await session.request('GET', url)
                    if resp.status == 200:
                        results[platform] = await resp.json()
                except Exception:
                    pass
            # Stylometry placeholder
            stylo = {"avg_word_length": 0.0, "vocabulary_richness": 0.0}
            return {
                'username': username,
                'profiles': results,
                'stylometry': stylo,
                'pgp_key_id': await self._find_pgp(username)
            }

    async def _find_pgp(self, username: str) -> Optional[str]:
        """Search keyserver for PGP key with username."""
        url = f"https://keyserver.ubuntu.com/pks/lookup?op=get&search={quote(username)}&options=mr"
        async with AsyncSession() as session:
            try:
                resp = await session.request('GET', url)
                html = await resp.text()
                match = re.search(r'pub\s+([A-F0-9]{16})', html)
                return match.group(1) if match else None
            except Exception:
                return None

class VulnIntelligence:

    async def search(self, keyword: str) -> List[Dict]:

            for vuln in data.get('vulnerabilities', []):

                })
            # Exploit-DB
            url2 = f"https://www.exploit-db.com/search?q={quote(keyword)}"
            # requires scraping; for demo we use a mock
            results.append({'source': 'ExploitDB', 'id': 'mock', 'description': 'See exploit-db.com for results'})
