# ============ TOOL 1: DEAD CONTENT RECOVERY ============
    async def research_dead_content(self, url: str) -> Dict[str, SnapshotInfo]:
        """
        Query multiple archives for dead content simultaneously.
        Uses Wayback CDX, Archive.today, Common Crawl, Memento, Google Cache.
        """
        results = {}
        
        async def check_wayback(target_url: str):
            try:
                async with self.session.get(
                    f"https://web.archive.org/cdx/search/cdx",
                    params={
                        'url': target_url,
                        'output': 'json',
                        'limit': 10,
                        'fl': 'timestamp,original,statuscode'
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        if len(data) > 1:  # Skip header row
                            snapshots = data[1:]  # First row is header
                            return [{
                                'source': 'Wayback',
                                'url': f"https://web.archive.org/web/{snap[0]}/{snap[1]}",
                                'timestamp': snap[0],
                                'status': snap[2]
                            } for snap in snapshots[:3]]
            except Exception as e:
                return [{'source': 'Wayback Error', 'url': '', 'timestamp': '', 'status': str(e)}]
        
        async def check_archive_today(target_url: str):
            try:
                async with self.session.get(
                    f"https://archive.ph/submit/",
                    params={'url': target_url}
                ) as response:
                    if response.status == 200:
                        text = await response.text()
                        match = re.search(r'https://archive\.ph/[^\s\'"]+', text)
                        if match:
                            return [{
                                'source': 'Archive.today',
                                'url': match.group(0),
                                'timestamp': datetime.now().isoformat(),
                                'status': '200'
                            }]
            except Exception as e:
                return [{'source': 'Archive.today Error', 'url': '', 'timestamp': '', 'status': str(e)}]
        
        async def check_common_crawl(target_url: str):
            try:
                # Common Crawl index API
                async with self.session.get(
                    f"https://index.commoncrawl.org/CC-MAIN-2024-33-index",
                    params={
                        'url': target_url,
                        'output': 'json',
                        'showNumPages': 'false'
                    }
                ) as response:
                    if response.status == 200:
                        text = await response.text()
                        lines = text.strip().split('\n')
                        snapshots = []
                        for line in lines[:3]:
                            try:
                                data = json.loads(line)
                                snapshots.append({
                                    'source': 'Common Crawl',
                                    'url': data.get('url', ''),
                                    'timestamp': data.get('timestamp', ''),
                                    'status': data.get('status', '')
                                })
                            except json.JSONDecodeError:
                                continue
                        return snapshots
            except Exception as e:
                return [{'source': 'Common Crawl Error', 'url': '', 'timestamp': '', 'status': str(e)}]
        
        async def check_memento(target_url: str):
            try:
                async with self.session.get(
                    f"http://timetravel.mementoweb.org/api/json/2024/{target_url}"
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        mementos = data.get('mementos', {}).get('list', [])
                        return [{
                            'source': 'Memento',
                            'url': m.get('uri', ''),
                            'timestamp': m.get('datetime', ''),
                            'status': '200'
                        } for m in mementos[:3]]
            except Exception as e:
                return [{'source': 'Memento Error', 'url': '', 'timestamp': '', 'status': str(e)}]
        
        async def check_google_cache(target_url: str):
            try:
                # Google cache via webcache.googleusercontent.com
                cache_url = f"http://webcache.googleusercontent.com/search?q=cache:{target_url}"
                async with self.session.get(cache_url, allow_redirects=True) as response:
                    if response.status == 200:
                        return [{
                            'source': 'Google Cache',
                            'url': cache_url,
                            'timestamp': datetime.now().isoformat(),
                            'status': str(response.status)
                        }]
            except Exception as e:
                return [{'source': 'Google Cache Error', 'url': '', 'timestamp': '', 'status': str(e)}]
        
        # Run all checks concurrently
        tasks = [
            check_wayback(url),
            check_archive_today(url),
            check_common_crawl(url),
            check_memento(url),
            check_google_cache(url)
        ]
        
        results_list = await asyncio.gather(*tasks)
        
        # Flatten results
        all_snapshots = []
        for snapshots in results_list:
            all_snapshots.extend(snapshots)
        
        return {'snapshots': all_snapshots}

# ============ TOOL 2: INVISIBLE WEB SCANNER ============
    async def research_invisible_web(self, domain: str) -> Dict[str, Any]:
        """
        Find hidden paths, check robots.txt, sitemap, and common sensitive files.
        Returns disallowed paths, 404 URLs, and hidden resources.
        """
        results = {
            'disallowed_paths': [],
            'sitemap_404s': [],
            'hidden_resources': [],
            'robots_txt': ''
        }
        
        # Ensure domain has protocol
        if not domain.startswith(('http://', 'https://')):
            domain = f"https://{domain}"
        
        async def fetch_robots_txt():
            try:
                async with self.session.get(f"{domain}/robots.txt") as response:
                    if response.status == 200:
                        text = await response.text()
                        results['robots_txt'] = text
                        # Parse disallow paths
                        disallow_paths = re.findall(r'Disallow:\s*(/.*)', text, re.IGNORECASE)
                        for path in disallow_paths:
                            if path and path != '/':
                                check_url = f"{domain}{path}"
                                try:
                                    async with self.session.get(check_url) as resp:
                                        results['disallowed_paths'].append({
                                            'path': check_url,
                                            'reachable': resp.status < 400,
                                            'status_code': resp.status
                                        })
                                except:
                                    results['disallowed_paths'].append({
                                        'path': check_url,
                                        'reachable': False,
                                        'status_code': 0
                                    })
            except Exception as e:
                print(f"Error fetching robots.txt: {e}")
        
        async def fetch_sitemap():
            try:
                async with self.session.get(f"{domain}/sitemap.xml") as response:
                    if response.status == 200:
                        text = await response.text()
                        root = ET.fromstring(text)
                        # Find all URLs
                        for url_elem in root.iter('{http://www.sitemaps.org/schemas/sitemap/0.9}url'):
                            loc = url_elem.find('{http://www.sitemaps.org/schemas/sitemap/0.9}loc')
                            if loc is not None and loc.text:
                                try:
                                    async with self.session.get(loc.text) as resp:
                                        if resp.status == 404:
                                            results['sitemap_404s'].append({
                                                'url': loc.text,
                                                'status': resp.status
                                            })
                                except:
                                    pass
            except Exception as e:
                print(f"Error fetching sitemap: {e}")
        
        async def check_hidden_paths():
            hidden_paths = [
                '/.env', '/.git/config', '/.DS_Store', 
                '/admin', '/debug', '/status.json',
                '/wp-admin', '/.gitignore', '/config.php',
                '/backup', '/temp', '/.htaccess'
            ]
            for path in hidden_paths:
                try:
                    async with self.session.get(f"{domain}{path}") as response:
                        if response.status < 400:
                            results['hidden_resources'].append({
                                'path': f"{domain}{path}",
                                'status_code': response.status,
                                'content_preview': (await response.text())[:200]
                            })
                except:
                    pass
        
        await asyncio.gather(
            fetch_robots_txt(),
            fetch_sitemap(),
            check_hidden_paths()
        )
        
        return results

# ============ TOOL 3: JS INTEL EXTRACTION ============
    async def research_js_intel(self, target_url: str) -> JSBotInfo:
        """
        Download JS files and extract API keys, endpoints, feature flags, WebSocket URLs.
        """
        result = {
            'url': target_url,
            'api_keys': [],
            'internal_endpoints': [],
            'feature_flags': [],
            'websocket_urls': [],
            'js_files': []
        }
        
        # Regex patterns for API keys
        patterns = {
            'api_keys': {
                'aws_key': r'AKIA[0-9A-Z]{16}',
                'github_token': r'ghp_[0-9a-zA-Z]{36}',
                'bearer_token': r'Bearer\s+[A-Za-z0-9\-_\.]+',
                'stripe_key': r'sk_live_[0-9a-zA-Z]{24}',
                'slack_token': r'xox[baprs]-[0-9a-zA-Z\-]+',
                'google_api': r'AIza[0-9A-Za-z\-_]{35}',
                'sendgrid': r'SG\.[0-9A-Za-z\-_]+',
                'mailgun': r'key-[0-9a-zA-Z]{32}',
            },
            'internal_endpoints': r'(?:/api/|/v[0-9]+/|/internal/|/private/)[^\s"\'<>\{\}]*',
            'feature_flags': r'(?:feature|flag|feature_flag|experiment|enable)\s*[:=]\s*["\']?([a-zA-Z0-9_]+)["\']?',
            'websocket_urls': r'(?:wss?://|WebSocket\()["\']?([^\s"\'\)]+)',
        }
        
        async def find_js_files():
            """Find all JS files referenced in HTML."""
            try:
                async with self.session.get(target_url) as response:
                    if response.status == 200:
                        html = await response.text()
                        # Find script tags
                        script_pattern = re.compile(r'<script[^>]*src=["\']([^"\']+\.js[^"\']*)["\']', re.IGNORECASE)
                        matches = script_pattern.findall(html)
                        result['js_files'] = [urljoin(target_url, m) for m in matches]
                        
                        # Also find inline scripts
                        inline_pattern = re.compile(r'<script[^>]*>(.*?)</script>', re.DOTALL | re.IGNORECASE)
                        inline_scripts = inline_pattern.findall(html)
                        for script in inline_scripts:
                            if script.strip():
                                result['js_files'].append(f"Inline: {script[:100]}...")
                                
            except Exception as e:
                print(f"Error finding JS files: {e}")
        
        async def extract_from_js(js_url: str):
            """Extract secrets from a single JS file."""
            try:
                if js_url.startswith('Inline:'):
                    content = js_url[7:-3]  # Remove "Inline: " prefix and "..." suffix
                else:
                    async with self.session.get(js_url) as response:
                        if response.status != 200:
                            return
                        content = await response.text()
                
                # Extract API keys
                for key_type, pattern in patterns['api_keys'].items():
                    matches = re.findall(pattern, content)
                    for match in matches:
                        if match not in result['api_keys']:
                            result['api_keys'].append({
                                'type': key_type,
                                'key': match,
                                'source': js_url
                            })
                
                # Extract internal endpoints
                endpoint_matches = re.findall(patterns['internal_endpoints'], content)
                for endpoint in endpoint_matches:
                    clean_endpoint = endpoint.strip('"\' ,;')
                    if clean_endpoint and clean_endpoint not in [e['endpoint'] for e in result['internal_endpoints']]:
                        result['internal_endpoints'].append({
                            'endpoint': clean_endpoint,
                            'source': js_url
                        })
                
                # Extract feature flags
                flag_matches = re.findall(patterns['feature_flags'], content, re.IGNORECASE)
                for flag in flag_matches:
                    if flag and flag not in [f['flag'] for f in result['feature_flags']]:
                        result['feature_flags'].append({
                            'flag': flag,
                            'source': js_url
                        })
                
                # Extract WebSocket URLs
                ws_matches = re.findall(patterns['websocket_urls'], content)
                for ws in ws_matches:
                    if ws and ws not in [w['url'] for w in result['websocket_urls']]:
                        result['websocket_urls'].append({
                            'url': ws,
                            'source': js_url
                        })
                        
            except Exception as e:
                print(f"Error extracting from {js_url}: {e}")
        
        # Find JS files first
        await find_js_files()
        
        # Extract from all JS files concurrently
        js_urls = [url for url in result['js_files'] if not url.startswith('Inline:')]
        inline_scripts = [url for url in result['js_files'] if url.startswith('Inline:')]
        
        tasks = [extract_from_js(url) for url in js_urls]
        if inline_scripts:
            tasks.append(extract_from_js(inline_scripts[0]))  # Process first inline script
        
        await asyncio.gather(*tasks)
        
        return result

    async def research_multi_search(self, query: str) -> List[SearchResult]:
        """
        Query DuckDuckGo + Brave + Marginalia + PublicWWW + Ahmia + crt.sh
        Deduplicate and rank results.
        """
        all_results = []
        seen_urls = set()
        
        async def search_duckduckgo():
            """Search DuckDuckGo Instant Answer API."""
            try:
                async with self.session.get(
                    "https://api.duckduckgo.com/",
                    params={
                        'q': query,
                        'format': 'json',
                        'no_html': '1',
                        'skip_disambig': '1'
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        # Extract from Results
                        for item in data.get('Results', []):
                            url = item.get('FirstURL', '')
                            if url and url not in seen_urls:
                                seen_urls.add(url)
                                all_results.append({
                                    'url': url,
                                    'title': item.get('Text', ''),
                                    'source': 'DuckDuckGo',
                                    'score': 0.9
                                })
                        # Extract from RelatedTopics
                        for topic in data.get('RelatedTopics', []):
                            url = topic.get('FirstURL', '')
                            if url and url not in seen_urls:
                                seen_urls.add(url)
                                all_results.append({
                                    'url': url,
                                    'title': topic.get('Text', ''),
                                    'source': 'DuckDuckGo',
                                    'score': 0.7
                                })
            except Exception as e:
                print(f"DuckDuckGo error: {e}")
        
        async def search_brave():
            """Search Brave Search API (free tier)."""
            try:
                headers = {
                    'Accept': 'application/json',
                    'Accept-Encoding': 'gzip'
                }
                async with self.session.get(
                    "https://api.search.brave.com/res/v1/web/search",
                    params={
                        'q': query,
                        'count': '10',
                        'country': 'US',
                        'search_lang': 'en'
                    },
                    headers=headers
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        for item in data.get('web', {}).get('results', []):
                            url = item.get('url', '')
                            if url and url not in seen_urls:
                                seen_urls.add(url)
                                all_results.append({
                                    'url': url,
                                    'title': item.get('title', ''),
                                    'source': 'Brave',
                                    'score': 0.85
                                })
            except Exception as e:
                print(f"Brave error: {e}")
        
        async def search_marginalia():
            """Search Marginalia Search API."""
            try:
                async with self.session.get(
                    "https://api.marginalia.nu/search",
                    params={
                        'query': query,
                        'format': 'json',
                        'count': '10'
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        for item in data.get('results', []):
                            url = item.get('url', '')
                            if url and url not in seen_urls:
                                seen_urls.add(url)
                                all_results.append({
                                    'url': url,
                                    'title': item.get('title', ''),
                                    'source': 'Marginalia',
                                    'score': 0.75
                                })
            except Exception as e:
                print(f"Marginalia error: {e}")
        
        async def search_publicww():
            """Search PublicWWW for source code."""
            try:
                # Note: PublicWWW requires API key, using alternative approach
                # Search via Google dork as fallback
                async with self.session.get(
                    "https://html.duckduckgo.com/html/",
                    params={'q': f'site:publicwww.com {query}'}
                ) as response:
                    if response.status == 200:
                        text = await response.text()
                        # Simple regex to extract URLs
                        url_pattern = re.compile(r'<a[^>]*href="([^"]+)"[^>]*>')
                        urls = url_pattern.findall(text)
                        for url in urls[:5]:
                            if url and url not in seen_urls and not url.startswith('/'):
                                seen_urls.add(url)
                                all_results.append({
                                    'url': url,
                                    'title': '',
                                    'source': 'PublicWWW',
                                    'score': 0.6
                                })
            except Exception as e:
                print(f"PublicWWW error: {e}")
        
        async def search_ahmia():
            """Search Ahmia for Tor hidden services."""
            try:
                async with self.session.get(
                    "https://ahmia.fi/search/",
                    params={'q': query}
                ) as response:
                    if response.status == 200:
                        text = await response.text()
                        # Extract .onion URLs
                        onion_pattern = re.compile(r'https?://[a-z0-9]{16}\.onion')
                        onions = onion_pattern.findall(text)
                        for onion in onions[:5]:
                            if onion not in seen_urls:
                                seen_urls.add(onion)
                                all_results.append({
                                    'url': onion,
                                    'title': '',
                                    'source': 'Ahmia',
                                    'score': 0.5
                                })
            except Exception as e:
                print(f"Ahmia error: {e}")
        
        async def search_crtsh():
            """Search crt.sh for certificates related to query."""
            try:
                # Extract potential domains from query
                domains = re.findall(r'[a-zA-Z0-9.-]+\.[a-zA-Z]{2,}', query)
                if domains:
                    async with self.session.get(
                        f"https://crt.sh/?q={domains[0]}",
                        headers={'Accept': 'application/json'}
                    ) as response:
                        if response.status == 200:
                            try:
                                data = await response.json()
                                for cert in data[:10]:
                                    name = cert.get('name_value', '')
                                    url = f"https://{name}" if not name.startswith('http') else name
                                    if url not in seen_urls and '.' in name:
                                        seen_urls.add(url)
                                        all_results.append({
                                            'url': url,
                                            'title': f"Certificate: {name}",
                                            'source': 'crt.sh',
                                            'score': 0.4
                                        })
                            except json.JSONDecodeError:
                                pass
            except Exception as e:
                print(f"crt.sh error: {e}")
        
        # Run all searches concurrently
        await asyncio.gather(
            search_duckduckgo(),
            search_brave(),
            search_marginalia(),
            search_publicww(),
            search_ahmia(),
            search_crtsh()
        )
        
        # Deduplicate and rank
        ranked_results = sorted(all_results, key=lambda x: x['score'], reverse=True)
        
        # Merge duplicate URLs (keep highest score)
        merged_results = []
        seen = set()
        for result in ranked_results:
            if result['url'] not in seen:
                seen.add(result['url'])
                merged_results.append(result)
        
        return merged_results[:20]  # Return top 20 results

# ============ TOOL 5: PASSIVE RECONNAISSANCE ============
    async def research_passive_recon(self, domain: str) -> Dict[str, Any]:
        """
        Gather subdomains, reverse IP, DNS records, and tech stack without direct scanning.
        """
        result = {
            'subdomains': [],
            'reverse_ip_hosts': [],
            'dns_records': {},
            'tech_stack': {},
            'http_headers': {}
        }
        
        async def get_subdomains_crtsh():
            """Get subdomains from crt.sh certificate transparency logs."""
            try:
                async with self.session.get(
                    f"https://crt.sh/?q=%25.{domain}&output=json"
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        seen = set()
                        for cert in data:
                            name = cert.get('name_value', '')
                            # Handle wildcard certificates
                            if name.startswith('*.'):
                                name = name[2:]
                            # Split multiple domains
                            names = name.split('\n') if '\n' in name else [name]
                            for n in names:
                                if n.endswith(domain) and n not in seen:
                                    seen.add(n)
                                    result['subdomains'].append({
                                        'subdomain': n,
                                        'source': 'crt.sh'
                                    })
            except Exception as e:
                print(f"crt.sh subdomain error: {e}")
        
        async def get_reverse_ip():
            """Get reverse IP hosts from hackertarget."""
            try:
                # First resolve domain to IP
                async with self.session.get(
                    f"https://api.hackertarget.com/reverseiplookup/?q={domain}"
                ) as response:
                    if response.status == 200:
                        text = await response.text()
                        hosts = text.strip().split('\n')
                        for host in hosts[1:]:  # Skip header
                            if host.strip():
                                result['reverse_ip_hosts'].append(host.strip())
            except Exception as e:
                print(f"Reverse IP error: {e}")
        
        async def get_dns_records():
            """Get DNS records using Google DNS over HTTPS."""
            dns_types = ['A', 'AAAA', 'MX', 'TXT', 'SPF', 'NS', 'SOA', 'CNAME', 'DKIM']
            
            async def query_dns(record_type):
                try:
                    # Use Google DNS-over-HTTPS
                    async with self.session.get(
                        f"https://dns.google/resolve",
                        params={
                            'name': domain,
                            'type': record_type
                        }
                    ) as response:
                        if response.status == 200:
                            data = await response.json()
                            answers = data.get('Answer', [])
                            records = []
                            for answer in answers:
                                records.append({
                                    'data': answer.get('data', ''),
                                    'ttl': answer.get('TTL', 0)
                                })
                            result['dns_records'][record_type] = records
                except Exception as e:
                    print(f"DNS {record_type} error: {e}")
            
            tasks = [query_dns(dns_type) for dns_type in dns_types]
            await asyncio.gather(*tasks)
        
        async def get_tech_stack():
            """Determine tech stack from HTTP headers and page content."""
            try:
                async with self.session.get(f"https://{domain}") as response:
                    headers = dict(response.headers)
                    result['http_headers'] = headers
                    
                    # Analyze headers for tech indicators
                    tech_indicators = {
                        'Server': headers.get('Server', ''),
                        'X-Powered-By': headers.get('X-Powered-By', ''),
                        'X-Generator': headers.get('X-Generator', ''),

                        'X-Magento-Cache': headers.get('X-Magento-Cache', ''),
                        'CF-Ray': headers.get('CF-Ray', '')  # Cloudflare
                    }
                    
                    # Detect CMS from headers
                    cms_patterns = {
                        'WordPress': ['wp-', 'wordpress'],
                        'Drupal': ['drupal'],
                        'Joomla': ['joomla'],
                        'Shopify': ['shopify'],
                        'Magento': ['magento'],
                        'ExpressJS': ['express'],

                        'ASP.NET': ['asp.net', 'x-aspnet-version'],
                        'Nginx': ['nginx'],
                        'Apache': ['apache'],
                        'Cloudflare': ['cloudflare'],
                    }
                    
                    detected = []
                    for tech, indicators in cms_patterns.items():
                        for indicator in indicators:
                            for header, value in tech_indicators.items():
                                if indicator.lower() in str(value).lower():

# ============ TOOL 6: IDENTITY RESOLUTION ============
    async def research_identity_resolve(self, email: str) -> IdentityInfo:
        """
        Resolve email to Gravatar, PGP key, and GitHub commits.
        """
        result = {
            'gravatar_hash': None,
            'pgp_key': None,
            'github_commits': []
        }
        
        async def get_gravatar():
            """Get Gravatar MD5 hash."""
            try:
                email_clean = email.strip().lower()
                md5_hash = hashlib.md5(email_clean.encode()).hexdigest()
                result['gravatar_hash'] = md5_hash
                
                # Check if Gravatar exists
                async with self.session.get(
                    f"https://www.gravatar.com/{md5_hash}.json"
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        result['gravatar_profile'] = data.get('entry', [{}])[0]
            except Exception as e:
                print(f"Gravatar error: {e}")
        
        async def get_pgp_key():
            """Search for PGP key on keys.openpgp.org."""
            try:
                async with self.session.get(
                    f"https://keys.openpgp.org/vks/v1/by-email/{email}"
                ) as response:
                    if response.status == 200:
                        # The response is the PGP key itself
                        key_text = await response.text()
                        # Extract key ID
                        match = re.search(r'Key ID:\s*([0-9A-Fa-f]+)', key_text)
                        if match:
                            result['pgp_key'] = {
                                'key_id': match.group(1),
                                'key_data': key_text[:500]  # First 500 chars
                            }
            except Exception as e:
                print(f"PGP key error: {e}")
        
        async def get_github_commits():
            """Extract email from GitHub commits."""
            try:
                # Extract username from email
                username = email.split('@')[0]
                
                # Search GitHub for commits by this email
                async with self.session.get(
                    f"https://api.github.com/search/commits",
                    headers={'Accept': 'application/vnd.github.cloak-preview'},
                    params={
                        'q': f'author-email:{email}',
                        'sort': 'author-date',
                        'per_page': 10
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        commits = []
                        for item in data.get('items', []):
                            commits.append({
                                'repo': item.get('repository', {}).get('full_name', ''),
                                'commit_hash': item.get('sha', ''),
                                'commit_message': item.get('commit', {}).get('message', '')[:100],
                                'date': item.get('commit', {}).get('author', {}).get('date', '')
                            })
                        result['github_commits'] = commits
            except Exception as e:
                print(f"GitHub commits error: {e}")
        
        await asyncio.gather(
            get_gravatar(),
            get_pgp_key(),
            get_github_commits()
        )
        
        return result

# ============ TOOL 7: CHANGE MONITORING ============
    async def research_change_monitor(self, url: str, db_path: str = 'changes.db') -> Optional[ChangeInfo]:
        """
        Monitor web pages for changes. Stores hashes in SQLite and diffs on re-check.
        """
        # Initialize SQLite database
        conn = sqlite3.connect(db_path)
        cursor = conn.cursor()
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS page_hashes (
                url TEXT PRIMARY KEY,
                content_hash TEXT,
                content TEXT,
                last_checked TIMESTAMP
            )
        ''')
        conn.commit()
        
        async def fetch_and_hash():
            """Fetch page content and compute hash."""
            try:
                async with self.session.get(url) as response:
                    if response.status == 200:
                        content = await response.text()
                        # Normalize content (remove whitespace for meaningful diffs)
                        normalized = re.sub(r'\s+', ' ', content).strip()
                        content_hash = hashlib.sha256(normalized.encode()).hexdigest()
                        return content_hash, content
            except Exception as e:
                print(f"Fetch error: {e}")
            return None, None
        
        # Fetch current content
        current_hash, current_content = await fetch_and_hash()
        if not current_hash:
            return None
        
        # Check for previous hash
        cursor.execute('SELECT content_hash, content FROM page_hashes WHERE url = ?', (url,))
        row = cursor.fetchone()
        
        if row:
            old_hash, old_content = row
            
            if current_hash != old_hash:
                # Changes detected - compute diff
                diff_lines = []
                old_lines = (old_content or '').split('\n')
                new_lines = current_content.split('\n')
                
                # Simple line-based diff
                old_set = set(old_lines)
                new_set = set(new_lines)
                
                added = new_set - old_set
                removed = old_set - new_set
                
                diff_lines.append("=== CHANGES DETECTED ===")
                if added:
                    diff_lines.append("Added lines:")
                    for line in list(added)[:10]:
                        diff_lines.append(f"  + {line[:100]}")
                if removed:
                    diff_lines.append("Removed lines:")
                    for line in list(removed)[:10]:
                        diff_lines.append(f"  - {line[:100]}")
                
                # Update database
                cursor.execute('''
                    UPDATE page_hashes 
                    SET content_hash = ?, content = ?, last_checked = datetime('now')
                    WHERE url = ?
                ''', (current_hash, current_content, url))
                conn.commit()
                
                return {
                    'url': url,
                    'hash_old': old_hash,
                    'hash_new': current_hash,
                    'diff': '\n'.join(diff_lines)
                }
            else:
                # No changes
                cursor.execute('''
                    UPDATE page_hashes SET last_checked = datetime('now') WHERE url = ?
                ''', (url,))
                conn.commit()
                return None
        else:
            # First time checking
            cursor.execute('''
                INSERT INTO page_hashes (url, content_hash, content, last_checked)
                VALUES (?, ?, ?, datetime('now'))
            ''', (url, current_hash, current_content))
            conn.commit()
            return None
        
        finally:
            conn.close()

# ============ TOOL 8: METADATA FORENSICS ============
    async def research_metadata_forensics(self, resource_url: str) -> MetadataInfo:
        """
        Extract EXIF from images, PDF metadata, and OpenGraph/JSON-LD from HTML.
        """
        result = {
            'exif': {},
            'pdf_metadata': {},
            'opengraph': {},
            'json_ld': {}
        }
        
        async def extract_exif():
            """Extract EXIF data from image URLs."""
            try:
                async with self.session.get(resource_url) as response:
                    if response.status == 200:
                        content = await response.read()
                        # Check if it's an image by MIME type
                        content_type = response.headers.get('Content-Type', '')
                        if 'image' in content_type:
                            try:
                                from PIL import Image
                                # Use Pillow to get EXIF
                                import io
                                img = Image.open(io.BytesIO(content))
                                exif_data = img._getexif()
                                if exif_data:
                                    result['exif'] = {
                                        str(k): str(v) for k, v in exif_data.items()
                                        if k in [271, 272, 274, 296, 305, 306, 315, 316, 318, 33434]
                                    }
                                # Basic image info
                                result['exif']['dimensions'] = f"{img.size[0]}x{img.size[1]}"
                                result['exif']['format'] = img.format
                                result['exif']['mode'] = img.mode
                            except ImportError:
                                # Fallback: extract basic info from bytes
                                result['exif']['info'] = f"Image detected, size: {len(content)} bytes"
            except Exception as e:
                print(f"EXIF error: {e}")
        
        async def extract_pdf_metadata():
            """Extract metadata from PDF files."""
            try:
                async with self.session.get(resource_url) as response:
                    if response.status == 200:
                        content = await response.read()
                        content_type = response.headers.get('Content-Type', '')
                        if 'pdf' in content_type or resource_url.endswith('.pdf'):
                            try:
                                import PyPDF2
                                import io
                                pdf_file = io.BytesIO(content)
                                reader = PyPDF2.PdfReader(pdf_file)
                                
                                metadata = reader.metadata
                                if metadata:
                                    result['pdf_metadata'] = {
                                        'title': metadata.get('/Title', ''),
                                        'author': metadata.get('/Author', ''),
                                        'subject': metadata.get('/Subject', ''),
                                        'creator': metadata.get('/Creator', ''),
                                        'producer': metadata.get('/Producer', ''),
                                        'creation_date': metadata.get('/CreationDate', ''),
                                        'modification_date': metadata.get('/ModDate', ''),
                                        'pages': len(reader.pages)
                                    }
                                
                                # Extract text from first page for analysis
                                if len(reader.pages) > 0:
                                    first_page = reader.pages[0]
                                    text = first_page.extract_text()[:500]
                                    result['pdf_metadata']['first_page_text'] = text
                                    
                            except ImportError:
                                # Extract basic info from bytes
                                result['pdf_metadata']['info'] = f"PDF detected, size: {len(content)} bytes"
            except Exception as e:
                print(f"PDF metadata error: {e}")
        
        async def extract_opengraph_and_jsonld():
            """Extract OpenGraph and JSON-LD from HTML pages."""
            try:
                async with self.session.get(resource_url) as response:
                    if response.status == 200:
                        html = await response.text()
                        content_type = response.headers.get('Content-Type', '')
                        
                        if 'html' in content_type or not any(ext in resource_url for ext in ['.pdf', '.jpg', '.png', '.gif']):
                            # Extract OpenGraph meta tags
                            og_patterns = {
                                'title': r'<meta[^>]*property=["\']og:title["\'][^>]*content=["\']([^"\']+)',
                                'description': r'<meta[^>]*property=["\']og:description["\'][^>]*content=["\']([^"\']+)',
                                'image': r'<meta[^>]*property=["\']og:image["\'][^>]*content=["\']([^"\']+)',
                                'url': r'<meta[^>]*property=["\']og:url["\'][^>]*content=["\']([^"\']+)',
                                'type': r'<meta[^>]*property=["\']og:type["\'][^>]*content=["\']([^"\']+)',
                                'site_name': r'<meta[^>]*property=["\']og:site_name["\'][^>]*content=["\']([^"\']+)',
                            }
                            
                            for key, pattern in og_patterns.items():
                                match = re.search(pattern, html, re.IGNORECASE)
                                if match:
                                    result['opengraph'][key] = match.group(1)
                            
                            # Extract Twitter Card meta tags
                            twitter_patterns = {
                                'twitter_card': r'<meta[^>]*name=["\']twitter:card["\'][^>]*content=["\']([^"\']+)',
                                'twitter_site': r'<meta[^>]*name=["\']twitter:site["\'][^>]*content=["\']([^"\']+)',
                                'twitter_creator': r'<meta[^>]*name=["\']twitter:creator["\'][^>]*content=["\']([^"\']+)',
                            }
                            for key, pattern in twitter_patterns.items():
                                match = re.search(pattern, html, re.IGNORECASE)
                                if match:
                                    result['opengraph'][key] = match.group(1)
                            
                            # Extract JSON-LD
                            jsonld_pattern = re.compile(
                                r'<script[^>]*type=["\']application/ld\+json["\'][^>]*>(.*?)</script>',
                                re.DOTALL | re.IGNORECASE
                            )
                            jsonld_matches = jsonld_pattern.findall(html)
                            if jsonld_matches:
                                try:
                                    result['json_ld'] = json.loads(jsonld_matches[0])
                                except json.JSONDecodeError:
                                    result['json_ld'] = {'error': 'Invalid JSON-LD'}
                            
                            # Extract schema.org microdata (simplified)
                            schema_pattern = re.compile(
                                r'itemscope[^>]*itemtype=["\']https?://schema\.org/([^"\']+)',
                                re.IGNORECASE
                            )
                            schemas = schema_pattern.findall(html)
                            if schemas:
                                result['opengraph']['schema_types'] = list(set(schemas))
            
            except Exception as e:
                print(f"OpenGraph/JSON-LD error: {e}")
        
        await asyncio.gather(
            extract_exif(),
            extract_pdf_metadata(),
            extract_opengraph_and_jsonld()
        )
        
        return result

# ============ TOOL 9: SOCIAL GRAPH ANALYSIS ============
    async def research_social_graph(self, identifier: str) -> Dict[str, SocialGraphInfo]:
        """
        Build social graph from GitHub co-contributors, Reddit threads, Semantic Scholar co-authors.
        """
        result = {
            'github': {'node': identifier, 'connections': [], 'connection_type': 'github'},
            'reddit': {'node': identifier, 'connections': [], 'connection_type': 'reddit'},
            'semantic_scholar': {'node': identifier, 'connections': [], 'connection_type': 'semantic_scholar'}
        }
        
        async def github_co_contributors():
            """Find GitHub co-contributors from repos."""
            try:
                # Get user's public repos
                async with self.session.get(
                    f"https://api.github.com/users/{identifier}/repos",
                    params={'per_page': 10, 'sort': 'updated'}
                ) as response:
                    if response.status == 200:
                        repos = await response.json()
                        contributors = set()
                        
                        for repo in repos[:5]:
                            repo_name = repo.get('full_name', '')
                            # Get contributors for each repo
                            async with self.session.get(
                                f"https://api.github.com/repos/{repo_name}/contributors",
                                params={'per_page': 5}
                            ) as contrib_response:
                                if contrib_response.status == 200:
                                    contrib_data = await contrib_response.json()
                                    for contributor in contrib_data:
                                        login = contributor.get('login', '')
                                        if login and login != identifier:
                                            contributors.add(login)
                        
                        result['github']['connections'] = list(contributors)[:20]
            except Exception as e:
                print(f"GitHub co-contributors error: {e}")
        
        async def reddit_comment_threads():
            """Find Reddit comment threads and interactions."""
            try:
                # Search for user's Reddit activity
                async with self.session.get(
                    f"https://www.reddit.com/user/{identifier}/comments.json",
                    headers={'User-Agent': 'ResearchBot/1.0'}
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        threads = data.get('data', {}).get('children', [])
                        
                        # Extract subreddits and interactions
                        subreddits = set()
                        users = set()
                        
                        for thread in threads[:20]:
                            thread_data = thread.get('data', {})
                            subreddit = thread_data.get('subreddit', '')
                            if subreddit:
                                subreddits.add(subreddit)
                            
                            # Get thread for interactions
                            thread_id = thread_data.get('link_id', '')
                            if thread_id:
                                async with self.session.get(
                                    f"https://www.reddit.com/r/{subreddit}/comments/{thread_id.replace('t3_', '')}.json",
                                    headers={'User-Agent': 'ResearchBot/1.0'}
                                ) as thread_response:
                                    if thread_response.status == 200:
                                        thread_data = await thread_response.json()
                                        # Extract all comment authors
                                        comments = thread_data[1].get('data', {}).get('children', [])
                                        for comment in comments[:10]:
                                            author = comment.get('data', {}).get('author', '')
                                            if author and author != identifier and author != '[deleted]':
                                                users.add(f"u/{author}")
                        
                        result['reddit']['connections'] = list(users)[:20]
                        result['reddit']['subreddits'] = list(subreddits)[:10]
            except Exception as e:
                print(f"Reddit error: {e}")
        
        async def semantic_scholar_coauthors():
            """Find co-authors from Semantic Scholar."""
            try:
                # Search for author by name
                async with self.session.get(
                    "https://api.semanticscholar.org/graph/v1/author/search",
                    params={
                        'query': identifier,
                        'limit': 3,
                        'fields': 'name,papers.author'
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        authors = data.get('data', [])
                        
                        if authors:
                            author_id = authors[0].get('authorId', '')
                            if author_id:
                                # Get detailed author info
                                async with self.session.get(
                                    f"https://api.semanticscholar.org/graph/v1/author/{author_id}",
                                    params={
                                        'fields': 'papers,authors',
                                        'limit': 50
                                    }
                                ) as author_response:
                                    if author_response.status == 200:
                                        author_data = await author_response.json()
                                        papers = author_data.get('papers', [])
                                        
                                        # Extract co-authors
                                        coauthors = set()
                                        for paper in papers[:10]:
                                            paper_authors = paper.get('authors', [])
                                            for author in paper_authors[:10]:
                                                name = author.get('name', '')
                                                if name and name != identifier:
                                                    coauthors.add(name)
                                        
                                        result['semantic_scholar']['connections'] = list(coauthors)[:20]
            except Exception as e:
                print(f"Semantic Scholar error: {e}")
        
        # Run all social graph tasks
        await asyncio.gather(
            github_co_contributors(),
            reddit_comment_threads(),
            semantic_scholar_coauthors()
        )
        
        return result

    async def research_competitive_intel(self, company: str) -> CompetitiveIntel:

        """
        result = {
            'edgar_filings': [],
            'patents': [],
            'github_activity': [],
            'certificate_changes': []
        }
        
        async def search_edgar():
            """Search SEC EDGAR for company filings."""
            try:
                # Search for company in EDGAR
                async with self.session.get(
                    f"https://efts.sec.gov/LATEST/search-index",
                    params={
                        'q': company,
                        'counts': '10',
                        'start': '0'
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        filings = data.get('hits', {}).get('hits', [])
                        
                        for filing in filings[:10]:
                            source = filing.get('_source', {})
                            result['edgar_filings'].append({
                                'company': source.get('company_name', ''),
                                'filing_type': source.get('form_type', ''),
                                'filing_date': source.get('filing_date', ''),
                                'cik': source.get('cik', ''),
                                'url': f"https://www.sec.gov/cgi-bin/browse-edgar?action=getcompany&CIK={source.get('cik', '')}"
                            })
            except Exception as e:
                print(f"EDGAR error: {e}")
        
        async def search_patents():
            """Search USPTO for patents."""
            try:
                # Using USPTO Patent Public Search API (free)
                async with self.session.get(
                    f"https://api.patentsview.org/patents/query",
                    params={
                        'q': f'{{"_text_any":{{"patent_title":"{company}"}}}}',
                        'f': 'patent_id,patent_title,patent_date,inventor_last_name'
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        patents = data.get('patents', [])
                        
                        for patent in patents[:10]:
                            result['patents'].append({
                                'id': patent.get('patent_id', ''),
                                'title': patent.get('patent_title', ''),
                                'date': patent.get('patent_date', ''),
                                'inventors': [inv.get('inventor_last_name', '') for inv in patent.get('inventors', [])]
                            })
            except Exception as e:
                print(f"Patent error: {e}")
        
        async def search_github_activity():
            """Search GitHub for company repositories and activity."""
            try:
                # Search for company repos
                async with self.session.get(
                    f"https://api.github.com/search/repositories",
                    params={
                        'q': f'{company} in:name,description',
                        'sort': 'updated',
                        'per_page': 10
                    }
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        repos = data.get('items', [])
                        
                        for repo in repos:
                            # Get recent commits
                            async with self.session.get(
                                f"https://api.github.com/repos/{repo.get('full_name', '')}/commits",
                                params={'per_page': 5}
                            ) as commit_response:
                                if commit_response.status == 200:
                                    commits = await commit_response.json()
                                    commit_data = []
                                    for commit in commits[:3]:
                                        commit_data.append({
                                            'sha': commit.get('sha', '')[:8],
                                            'message': commit.get('commit', {}).get('message', '')[:100],
                                            'author': commit.get('commit', {}).get('author', {}).get('name', ''),
                                            'date': commit.get('commit', {}).get('author', {}).get('date', '')
                                        })
                                    
                                    result['github_activity'].append({
                                        'repo': repo.get('full_name', ''),
                                        'description': repo.get('description', ''),
                                        'stars': repo.get('stargazers_count', 0),
                                        'forks': repo.get('forks_count', 0),
                                        'language': repo.get('language', ''),
                                        'recent_commits': commit_data
                                    })
            except Exception as e:
                print(f"GitHub activity error: {e}")
        
        async def monitor_certificates():
            """Monitor certificate transparency logs for domain changes."""
            try:
                # Extract domain from company name
                # This is simplified - in production you'd use domain discovery
                domain = f"{company.lower().replace(' ', '')}.com"
                
                # Get certificates from crt.sh
                async with self.session.get(
                    f"https://crt.sh/?q=%25.{domain}&output=json"
                ) as response:
                    if response.status == 200:
                        data = await response.json()
                        
                        # Group certificates by issuer
                        issuers = {}
                        for cert in data[:50]:
                            issuer = cert.get('issuer_name', 'Unknown')
                            if issuer not in issuers:
                                issuers[issuer] = []
                            issuers[issuer].append({
                                'id': cert.get('id', ''),
                                'name_value': cert.get('name_value', ''),
                                'not_before': cert.get('not_before', ''),
                                'not_after': cert.get('not_after', ''),
                                'issuer': issuer
                            })
                        
                        # Get latest certificates for each issuer
                        for issuer, certs in issuers.items():
                            certs.sort(key=lambda x: x.get('not_before', ''), reverse=True)
                            result['certificate_changes'].extend(certs[:3])
            except Exception as e:
                print(f"Certificate monitoring error: {e}")
        
        # Run all competitive intelligence tasks
        await asyncio.gather(
            search_edgar(),
            search_patents(),
            search_github_activity(),
            monitor_certificates()
        )
        
        return result
