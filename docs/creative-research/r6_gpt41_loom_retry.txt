import aiohttp

# Tool 1: Dead Content Finder
class DeadContentResult(TypedDict):
    cdx: Optional[List[dict]]
    google_cache: Optional[str]
    archive_today: Optional[List[str]]
    common_crawl: Optional[List[str]]
    memento: Optional[List[str]]
    error: Optional[str]

async def dead_content_finder(url: str, session: aiohttp.ClientSession) -> DeadContentResult:
    result: DeadContentResult = {"cdx": None, "google_cache": None, "archive_today": None,
                                 "common_crawl": None, "memento": None, "error": None}
    try:
        # CDX (Wayback Machine)
        cdx_url = f"http://web.archive.org/cdx/search/cdx?url={url}&output=json&limit=5"
        async with session.get(cdx_url, timeout=7) as r:
            if r.status == 200:
                j = await r.json()
                if len(j) > 1:
                    keys = j[0]
                    records = [dict(zip(keys, x)) for x in j[1:]]
                    result["cdx"] = records

        # Google Cache (SERP preview link; actual HTML is not directly accessible via API)
        gcache_url = f"https://webcache.googleusercontent.com/search?q=cache:{url}"
        async with session.get(gcache_url, timeout=7) as r:
            if r.status == 200 and "Google Cache" in await r.text():
                result["google_cache"] = gcache_url

        # Archive.today
        atoday_url = f"https://archive.today/submit/?url={url}"
        async with session.get(atoday_url, timeout=7) as r:
            text = await r.text()
            matches = re.findall(r"//archive.today/\d+/\w+", text)
            result["archive_today"] = [f"https:{m}" for m in matches] if matches else []

        # Common Crawl (index API)
        cc_url = f"http://index.commoncrawl.org/CC-MAIN-2023-50-index?url={url}&output=json"
        async with session.get(cc_url, timeout=7) as r:
            if r.status == 200:
                lines = (await r.text()).splitlines()
                result["common_crawl"] = [json.loads(line)["filename"] for line in lines if line]

        # Memento aggregator
        memento_url = f"http://memgator.cs.odu.edu/timemap/json/{url}"
        async with session.get(memento_url, timeout=7) as r:
            if r.status == 200:
                j = await r.json()
                result["memento"] = list(j.get("mementos", {}).get("list", []))

    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 2: Hidden Web Explorer
class HiddenWebResult(TypedDict):
    robots_disallowed: List[str]
    sitemap_404s: List[str]
    exposures: Dict[str, List[str]]
    error: Optional[str]

async def hidden_web_explorer(domain: str, session: aiohttp.ClientSession) -> HiddenWebResult:
    result: HiddenWebResult = {"robots_disallowed": [], "sitemap_404s": [], "exposures": {}, "error": None}
    try:
        robots_url = f"https://{domain}/robots.txt"
        async with session.get(robots_url, timeout=7) as r:
            if r.status == 200:
                robots_txt = await r.text()
                disallowed = [line.split(":")[1].strip() for line in robots_txt.splitlines()
                              if line.lower().startswith('disallow')]
                result["robots_disallowed"] = disallowed

        # Sitemaps
        sitemap_urls = [f"https://{domain}/sitemap.xml", f"http://{domain}/sitemap.xml"]
        for s_url in sitemap_urls:
            try:
                async with session.get(s_url, timeout=7) as r:
                    if r.status == 404:
                        result["sitemap_404s"].append(s_url)
            except Exception:
                continue

        # Exposed secrets and sourcemap detection
        exposures = {}
        test_paths = [".env", ".git/config", "config.js", "webpack.config.js", "main.js.map", "bundle.js.map"]
        for path in test_paths:
            url = f"https://{domain}/{path}"
            try:
                async with session.get(url, timeout=7) as r:
                    if r.status == 200:
                        exposures.setdefault(path, []).append(url)
            except Exception:
                continue
        result["exposures"] = exposures

    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 3: JS Intelligence
class JSIntelResult(TypedDict):
    api_keys: Dict[str, List[str]]
    endpoints: List[str]
    feature_flags: List[str]
    error: Optional[str]

async def js_intelligence(js_url: str, session: aiohttp.ClientSession) -> JSIntelResult:
    result: JSIntelResult = {"api_keys": {}, "endpoints": [], "feature_flags": [], "error": None}
    try:
        async with session.get(js_url, timeout=10) as r:
            if r.status != 200:
                result["error"] = f"Failed to fetch {js_url}: status {r.status}"
                return result
            js = await r.text()
            # API key patterns (generic)
            key_patterns = {
                "Google": r'AIza[0-9A-Za-z-_]{35}',
                "AWS": r'AKIA[0-9A-Z]{16}',
                "Slack": r'xox[baprs]-([0-9a-zA-Z]{10,48})?',
                "Mailgun": r'key-[0-9a-zA-Z]{32}',
                "Stripe": r'sk_live_[0-9a-zA-Z]{24}',
            }
            for name, pat in key_patterns.items():
                matches = re.findall(pat, js)
                if matches:
                    result["api_keys"][name] = matches

            # Endpoint discovery (api/http URLs)
            endpoints = re.findall(r'https?://[^\s\"\'<>\\]+', js)
            result["endpoints"] = list(set(endpoints))

            # Feature flags (simple: look for "flag" keys)
            flags = re.findall(r'([A-Za-z0-9_]+_flag)\s*[:=]\s*(true|false|"true"|\'true\'|"false"|\'false\')', js, re.I)
            result["feature_flags"] = [flag for flag, _ in flags]
    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 4: Multi-Engine Search (20+)
class SearchResult(TypedDict):
    engine: str
    title: str
    link: str
    snippet: str

class MultiSearchResult(TypedDict):
    results: List[SearchResult]
    error: Optional[str]

SEARCH_ENGINES = {
    # Engine: Tuple("API endpoint", lambda to extract results)
    "DuckDuckGo": (
        "https://api.duckduckgo.com/?q={q}&format=json",
        lambda j: [{"engine": "DuckDuckGo", "title": r.get('Text', ''), "link": r.get('FirstURL', ''), "snippet": r.get("Text", "")}
                   for r in j.get("RelatedTopics", []) if r.get('FirstURL')],
    ),
    "Wikipedia": (
        "https://en.wikipedia.org/w/api.php?action=query&format=json&list=search&srsearch={q}",
        lambda j: [{"engine": "Wikipedia", "title": s['title'], "link": f"https://en.wikipedia.org/wiki/{s['title'].replace(' ', '_')}", "snippet": s["snippet"]}
                   for s in j.get("query", {}).get("search", [])],
    ),
    # Add more engines here (Bing, Qwant, Mojeek, etc. But most require keys)
    "ContextualWeb": (
        "https://contextualwebsearch-websearch-v1.p.rapidapi.com/api/Search/WebSearchAPI?q={q}&pageNumber=1&pageSize=5&autoCorrect=true",
        lambda j: [{"engine": "ContextualWeb", "title": v["title"], "link": v["url"], "snippet": v["snippet"]} for v in j.get("value", [])]
    )
}
# Free engines generally only: for some API you may need to request a free RapidAPI key and set an environment variable.

async def multi_engine_search(query: str, session: aiohttp.ClientSession) -> MultiSearchResult:
    results = []
    dedup: Set[str] = set()
    errors = []

    headers = {"x-rapidapi-key": "SIGN-UP-FOR-KEY", "x-rapidapi-host": "contextualwebsearch-websearch-v1.p.rapidapi.com"}  # For ContextualWeb; user must add key.

    async def fetch(engine: str, e_data: tuple):
        url, extractor = e_data
        u = url.format(q=query)
        try:
            h = headers if "contextualweb" in u else {}
            async with session.get(u, timeout=10, headers=h) as r:
                if r.status != 200:
                    errors.append(f"{engine} status {r.status}")
                    return
                j = await r.json()
                rlist = extractor(j)
                for r in rlist:
                    k = r["link"]
                    if k and k not in dedup:
                        dedup.add(k)
                        results.append(r)
        except Exception as e:
            errors.append(f"{engine} err: {e}")

    tasks = [fetch(engine, e_data) for engine, e_data in SEARCH_ENGINES.items()]
    await asyncio.gather(*tasks)
    return MultiSearchResult(results=results, error="; ".join(errors) if errors else None)

# Tool 5: Passive Recon
class PassiveReconResult(TypedDict):
    crtsh: List[str]
    reverse_ip: List[str]
    dns_records: Dict[str, Any]
    http_headers: Dict[str, str]
    error: Optional[str]

async def passive_recon(domain: str, session: aiohttp.ClientSession) -> PassiveReconResult:
    result: PassiveReconResult = {"crtsh": [], "reverse_ip": [], "dns_records": {}, "http_headers": {}, "error": None}
    try:
        # CT logs crt.sh
        crtsh_url = f"https://crt.sh/?q={domain}&output=json"
        async with session.get(crtsh_url, timeout=10) as r:
            if r.status == 200:
                try:
                    entries = await r.json()
                    result["crtsh"] = list({e.get("name_value") for e in entries if e.get("name_value")})
                except Exception:
                    result["crtsh"] = []

        # Reverse IP at hackertarget
        rip_url = f"https://api.hackertarget.com/reverseiplookup/?q={domain}"
        async with session.get(rip_url, timeout=10) as r:
            text = await r.text()
            if "error" not in text.lower():
                result["reverse_ip"] = [line.strip() for line in text.splitlines() if "." in line]

        # DNS records via aiodns (or via hackertarget again)
        resolver = aiodns.DNSResolver()
        try:
            a_cl = await resolver.gethostbyname(domain, socket.AF_INET)
            result["dns_records"]["A"] = a_cl.addresses
        except Exception:
            pass

        # HTTP headers
        try:
            async with session.get(f'https://{domain}', timeout=7) as r:
                result["http_headers"] = dict(r.headers)
        except Exception:
            result["http_headers"] = {}
    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 6: Identity Linker
class IdentityResult(TypedDict):
    gravatar: Optional[str]
    pgp_keys: List[str]
    github_emails: List[str]
    username_found: List[Dict[str, str]]
    error: Optional[str]

async def identity_linker(email: Optional[str]=None, username: Optional[str]=None, session: aiohttp.ClientSession=None) -> IdentityResult:
    result: IdentityResult = {"gravatar": None, "pgp_keys": [], "github_emails": [], "username_found": [], "error": None}
    try:
        # Gravatar
        if email:
            hash_email = md5(email.strip().lower().encode()).hexdigest()
            gravatar_url = f"https://www.gravatar.com/avatar/{hash_email}?d=404"
            async with session.get(gravatar_url, timeout=7) as r:
                if r.status == 200:
                    result["gravatar"] = gravatar_url

        # PGP (MIT server)
        if email:
            pgp_url = f"https://pgp.mit.edu/pks/lookup?op=vindex&search={email}"
            async with session.get(pgp_url, timeout=7) as r:
                txt = await r.text()
                keys = re.findall(r'Key ID ([0-9A-F]+)', txt)
                result["pgp_keys"] = keys

        # GitHub commit email (search commits for email)
        if email:
            g_url = f"https://api.github.com/search/commits?q=author-email:{email}"
            async with session.get(g_url, timeout=10, headers={"Accept": "application/vnd.github.cloak-preview"}) as r:
                if r.status == 200:
                    j = await r.json()
                    emails = []
                    for c in j.get("items", []):
                        author = c.get("commit", {}).get("author", {})
                        if author.get("email"):
                            emails.append(author.get("email"))
                    result["github_emails"] = emails

        # Username enumeration (social web)
        user_services = [
            ("Twitter", f"https://twitter.com/{username}"),
            ("GitHub", f"https://github.com/{username}"),
            ("Reddit", f"https://www.reddit.com/user/{username}"),
            ("Keybase", f"https://keybase.io/{username}"),
        ]
        if username:
            for site, url in user_services:
                try:
                    async with session.get(url, timeout=5) as r:
                        if r.status == 200:
                            result["username_found"].append({"site": site, "url": url})
                except Exception:
                    continue
    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 7: Change Detector
class ChangeDetectorResult(TypedDict):
    content_hash: Optional[str]
    has_changed: bool
    diff: Optional[str]
    semantic_change: Optional[str]
    error: Optional[str]

async def change_detector(url: str, session: aiohttp.ClientSession, db_path: str = "change_detector.db") -> ChangeDetectorResult:
    result: ChangeDetectorResult = {"content_hash": None, "has_changed": False, "diff": None, "semantic_change": None, "error": None}
    try:
        async with session.get(url, timeout=15) as r:
            text = await r.text()
            content_hash = md5(text.encode()).hexdigest()
            result["content_hash"] = content_hash

            # SQLite record
            conn = sqlite3.connect(db_path)
            c = conn.cursor()
            c.execute("CREATE TABLE IF NOT EXISTS snapshots (url TEXT PRIMARY KEY, content_hash TEXT, content TEXT)")
            c.execute("SELECT content_hash, content FROM snapshots WHERE url=?", (url,))
            row = c.fetchone()
            if row:
                old_hash, old_content = row
                result["has_changed"] = old_hash != content_hash
                result["diff"] = "\n".join([line for line in text.splitlines() if line not in old_content.splitlines()])
                # basic semantic change: did HTML title change?
                old_title = re.search(r"<title>(.*?)</title>", old_content, re.I)
                new_title = re.search(r"<title>(.*?)</title>", text, re.I)
                if old_title and new_title and old_title.group(1) != new_title.group(1):
                    result["semantic_change"] = "Title changed"
            else:
                result["has_changed"] = True

            # Upsert
            c.execute("REPLACE INTO snapshots(url,content_hash,content) VALUES (?, ?, ?)", (url, content_hash, text))
            conn.commit()
            conn.close()
    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 8: Metadata Extractor
class MetadataResult(TypedDict):
    exif: Dict[str, Any]
    pdfmeta: Dict[str, Any]
    jsonld: List[Dict[str, Any]]
    opengraph: Dict[str, Any]
    microdata: Dict[str, Any]
    error: Optional[str]

async def metadata_extractor(url: str, session: aiohttp.ClientSession) -> MetadataResult:
    import magic
    import exifread
    from PyPDF2 import PdfReader

    result: MetadataResult = {"exif": {}, "pdfmeta": {}, "jsonld": [], "opengraph": {}, "microdata": {}, "error": None}
    try:
        async with session.get(url, timeout=15) as r:
            content = await r.read()
            mime = magic.from_buffer(content, mime=True)

            # EXIF
            if mime.startswith("image"):
                try:
                    from io import BytesIO
                    img = BytesIO(content)
                    img.seek(0)
                    tags = exifread.process_file(img, details=False)
                    result["exif"] = {k: str(v) for k, v in tags.items()}
                except Exception:
                    result["exif"] = {}

            # PDF
            if mime == "application/pdf":
                try:
                    from io import BytesIO
                    pdf = PdfReader(BytesIO(content))
                    result["pdfmeta"] = dict(pdf.metadata)
                except Exception:
                    result["pdfmeta"] = {}

            # HTML-based metadata
            if mime.startswith("text/html"):
                soup = BeautifulSoup(content, "lxml")
                # JSON-LD
                jld = [json.loads(tag.text) for tag in soup.find_all("script", type="application/ld+json")]
                result["jsonld"] = jld

                # OpenGraph/meta
                opengraph = {}
                for tag in soup.find_all("meta"):
                    if tag.get("property", "").startswith("og:"):
                        opengraph[tag.get("property")] = tag.get("content", "")
                result["opengraph"] = opengraph

                # Microdata (itemscope/itemprop)
                micro = {}
                for scope in soup.find_all(attrs={"itemscope": True}):
                    props = {prop.get("itemprop"): prop.get("content", "") for prop in scope.find_all(attrs={"itemprop": True})}
                    micro.update(props)
                result["microdata"] = micro
    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 9: Social Graph
class SocialGraphResult(TypedDict):
    github_cocontributors: List[str]
    reddit_threads: List[str]
    coauthors: List[str]
    error: Optional[str]

async def social_graph(github_user: str, reddit_user: str, scholar_query: str, session: aiohttp.ClientSession) -> SocialGraphResult:
    result: SocialGraphResult = {"github_cocontributors": [], "reddit_threads": [], "coauthors": [], "error": None}
    try:
        # GitHub co-contributors (first repo starred)
        gh_url = f"https://api.github.com/users/{github_user}/repos"
        async with session.get(gh_url, timeout=10) as r:
            if r.status == 200:
                repos = await r.json()
                if repos:
                    repo = repos[0]
                    contributors_url = repo.get("contributors_url")
                    if contributors_url:
                        async with session.get(contributors_url, timeout=7) as rc:
                            cjson = await rc.json()
                            result["github_cocontributors"] = [c.get("login") for c in cjson if c.get("login", "") != github_user]

        # Reddit user comments/threads
        red_url = f"https://www.reddit.com/user/{reddit_user}/submitted.json"
        async with session.get(red_url, headers={"User-Agent": "Mozilla"}, timeout=10) as r:
            if r.status == 200:
                j = await r.json()
                result["reddit_threads"] = [t["data"]["title"] for t in j["data"]["children"] if t["kind"] == "t3"]

        # Academic co-authors (Semantic Scholar API, free)
        s2_url = f"https://api.semanticscholar.org/graph/v1/author/search?query={scholar_query}&fields=name,papers.authors"
        async with session.get(s2_url, timeout=10) as r:
            if r.status == 200:
                j = await r.json()
                coauthors = set()
                for author in j.get("data", [])[:1]:  # Take first match
                    for paper in author.get("papers", []):
                        for a in paper.get("authors", []):
                            if a.get("name") != author.get("name"):
                                coauthors.add(a.get("name"))
                result["coauthors"] = list(coauthors)
    except Exception as e:
        result["error"] = str(e)
    return result

# Tool 10: Competitive Intel
class CompetitiveIntelResult(TypedDict):
    sec_filings: List[str]
    patents: List[dict]
    github_activity: List[dict]
    dns_changes: List[str]
    error: Optional[str]

async def competitive_intel(company: str, github_org: str, domain: str, session: aiohttp.ClientSession) -> CompetitiveIntelResult:
    result: CompetitiveIntelResult = {"sec_filings": [], "patents": [], "github_activity": [], "dns_changes": [], "error": None}
    try:
        # SEC filings (EDGAR via sec-api)
        sec_url = f"https://www.sec.gov/cgi-bin/browse-edgar?action=getcompany&CIK={company}&type=&output=atom"
        async with session.get(sec_url, timeout=10) as r:
            if r.status == 200:
                feed = await r.text()
                links = re.findall(r"<link href='([^']+)'", feed)
                result["sec_filings"] = links

        # Patents - PatentsView (USPTO open API)
        pat_url = f"https://search.patentsview.org/api/v1/patents?q={{\"_text_any\":{{\"patent_title\":\"{company}\"}}}}&f=[\"patent_number\",\"patent_title\"]"
        async with session.get(pat_url, timeout=13) as r:
            if r.status == 200:
                pats = await r.json()
                result["patents"] = pats.get("patents", [])

        # GitHub activity (events API)
        gh_url = f"https://api.github.com/orgs/{github_org}/events"
        async with session.get(gh_url, timeout=10) as r:
            if r.status == 200:
                acts = await r.json()
                result["github_activity"] = acts

        # DNS changes (SecurityTrails sample/free DNS API, limited)
        st_url = f"https://api.securitytrails.com/v1/domain/history/{domain}/dns/a"
        async with session.get(st_url, headers={"APIKEY": "SIGN-UP-FOR-KEY"}, timeout=10) as r:
            if r.status == 200:
                j = await r.json()
                result["dns_changes"] = [h["ip"] for h in j.get("records", []) if "ip" in h]
    except Exception as e:
        result["error"] = str(e)
    return result
